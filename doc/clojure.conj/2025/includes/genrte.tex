(def rte-keys [:sigma :empty-set :epsilon
               :type :cat :* :and :or :not])

(defn gen-rte [depth]
  (let [key (rand-nth rte-keys) 
        g (fn [] (gen-rte (dec depth)))]
    (case key
      (:type) (rand-nth known-types)

      (:sigma :empty-set :epsilon) key

      (:and :or :cat)
        (list key (g) (g))

      (:* :not)
        (list key (g)))))
