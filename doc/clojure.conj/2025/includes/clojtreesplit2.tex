;; Gaussian split
(defn tree-split-rte-gaussian [leaves]
  (tree-split-rte leaves 
                  (fn [n]
                    (case n
                      (1 2 3) 1
                      (max 1 (+ (rand-int (quot n 2))
                                (rand-int (quot n 2))))))))

;; inverse Gaussian split
(defn tree-split-rte-inv-gaussian [leaves]
  (tree-split-rte leaves
                  (fn [n]
                    (let [r (max 1 (rand-int (quot n 2)))]
                       (if (= 0 (rand-int 2))
                         r
                         (- n r))))))
