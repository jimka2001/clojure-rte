001  (ns rte-randomize-syntax
002    (:require [genus.genus-tester :refer [*test-types*]])
003  )
004  
005  
006  (def ^:dynamic *rte-keywords*
007    [:type
008     :? :+ :* :not
009     :and :or 
010     :cat ;; :permute
011     :contains-any :contains-every :contains-none
012     :sigma :empty-set :epsilon])
013  
014  (defn gen-rte
015    ([depth]
016     (gen-rte depth *test-types*))
017    ([depth types]
018     (let [key (rand-nth *rte-keywords*)] 
019       (gen-rte key depth types)))
020    ([key depth types]
021     (case key
022       (:type) (rand-nth types)
023       (:sigma :empty-set :epsilon) key
024       ;;(:permute) (gen-rte :cat depth types)
025       (:and :or :cat :contains-any
026             :contains-every :contains-none) (cons key (map (fn [_k] (gen-rte (dec depth) types))
027                                                            (range depth)))
028       (:? :+ :* :not) (list key (gen-rte (dec depth) types)))))
