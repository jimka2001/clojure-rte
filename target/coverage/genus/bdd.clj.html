064  (defn dnf
065    "Serialize a Bdd to dnf disjunctive normal form.
066    This dnf form is cleaned up so that an (and ...) or (or ...) clause contains
067    no subtype/supertype pairs.  This subtype relation is determined by
068    (gns/subtype? a b false).  
069    "
070    [bdd]
071    (letfn [(exists-supertype [sub types]
072              (exists [super types]
073                      (c/and (not= sub super)
074                             (gns/subtype? sub super false))))
075            (check-supers [args]
076              (let [args (distinct args)
077                    complements (for [a args
078                                      b args
079                                      :when (c/or (= a (list 'not b))
080                                                  (= b (list 'not a)))]
081                                  [a b])]
082                (cond
083                  ;; does the list contain A and (not A) ?
084                  (non-empty? complements)
085                  '(:sigma)
086  
087                  ;; does the list contain A and B where A is subtype B
088                  :else
089                  (remove (fn [sub]
090                            (exists-supertype sub args))
091                          args))))]
092  
093      (gns/create-or
094       (check-supers
095        (call-with-collector
096         (fn [collect]
097           (letfn [(walk [node parents]
098                     (let [my-label (:label node)
099                           ;; two lazy sequences created by filter.  the filter loops are
100                           ;; never called unless node is differrent from true or false, i.e.
101                           ;; the 3rd case of the cond.
102                           exists-disjoint (delay (exists [x parents] (gns/disjoint? x my-label false)))
103                           exists-subtype (delay (exists [x parents] (gns/subtype?  x my-label false)))]
104                       (cond
105                         (= true node) ; case #1
106                         ;; we know parents ( ... A ... B ...) that B is not subtype of A, but maybe A subtype B
107                         ;;   we need to remove the supertypes
108                         ;;   E.g., (Long java.io.Comparable java.io.Serializable) -> (Long)
109                         ;;   E.g.  (Long Number) -> Long
110                         (let [term (loop [tail parents
111                                           done '()]
112                                      (if (empty? tail)
113                                        done
114                                        (let [keeping (remove (fn [b]
115                                                                ;; if we don't know, then keep it.  it might
116                                                                ;; be redunant, but it won't be wrong.
117                                                                ;; Is (first tail) <: b ?
118                                                                ;;   if yes, then omit b in recur call
119                                                                ;;   if :dont-know then keep it.
120                                                                (gns/subtype? (first tail) b false))
121                                                              (rest tail))]
122                                          (recur keeping
123                                                 (cons (first tail) done)))))]
124                           (collect (gns/create-and term)))
125                         
126                         (= false node) ; case #2
127                         nil ;; do not collect, and prune recursion
128  
129                         @exists-disjoint ; case #3
130                         ;; this means that the type-designators returned by (:label node)
131                         ;;    is disjoint with something in the parents list.
132                         ;;    E.g., (:label node)=String, and parents= ( ... Number ...).
133                         ;;    So we can  prune the positive branch, and omit this node from the
134                         ;;    parents list in the recursive call.  Why, because any object
135                         ;;    which has type of everything in the parents list is
136                         ;;    guaranteed to NOT have the type of node.   We could
137                         ;;    add `(not ~(label :node)) to the parents list, but it
138                         ;;    would be redundant---we would have (and (not String) ... Number ...),
139                         ;;    so we'll keep just (and ... Number ...).
140                         (walk (:negative node)
141                               parents)
142                         
143                         @exists-subtype ; case #4
144                         ;; this means that the type-designators returned by (:label node)
145                         ;;    is a supertype of something in the parents list.  E.g.,
146                         ;;    (:label node) = Number and parents= (... Long ...).
147                         ;;    Wo we can prune the negative branch, and omit this node from the
148                         ;;    parents list in the recursive call.  Why, because any object
149                         ;;    which has type of everything in the parents list is
150                         ;;    guaranteed to have the type of node.   We could
151                         ;;    add (label :node) to the parents list, but it
152                         ;;    would be redundant---we would have (and Number ... Long ...)
153                         ;;    so we'll keep just (and ... Long ...).
154                         (walk (:positive node)
155                               parents)
156                         
157                         :else ; case #5
158                         (do (walk (:positive node)
159                                   (cons (:label node) parents))
160                             (walk (:negative node)
161                                   (cons (list 'not (:label node)) parents))))))]
162             (walk bdd '()))))))))

282      false
283  
284      :else
285      (node type-designator true false)))

287  (defn label-<-by-index
288    "A strict ordering function on potential labels in a Bdd.
289    This function enforces that a < (satifies f) whenever
290    a is not a (satisfies g) type, but otherwise, a < b whenever
291    a was encountered first in the computation.  Whenever a new
292    label is discovered in generating a Bdd, it is given an index
293    higher than any other existing index.  The ordering of the labels
294    is the ordering of the index  index-of(a) < index-of(b)."
295    [t1 t2]
296    (let [label-index-1 (type-index t1)
297          label-index-2 (type-index t2)]
298      (assert (integer? label-index-1) (format "expecting integer got %s" (type label-index-1)))
299      (assert (integer? label-index-2) (format "expecting integer got %s" (type label-index-2)))
300      (cond (c/and (gns/satisfies? t1)
301                 (gns/satisfies? t2))
302            (< label-index-1 label-index-2)
303  
304            (gns/satisfies? t1)
305            false
306  
307            (gns/satisfies? t2)
308            true
309  
310            :else
311            (< label-index-1 label-index-2))))

374    "Perform a Boolean AND on 0 or more Bdds."
375    (fn
376      ([] true)
377      ([bdd] bdd)
378      ([bdd1 bdd2]
379       (cond
380         (= false bdd1) false
381         (= false bdd2) false
382         (= true bdd1) bdd2
383         (= true bdd2) bdd1
384         (identical? bdd1 bdd2) bdd1
385         :else (binary-op bdd/and bdd1 bdd2)))
386      ([bdd1 bdd2 & bdds]
387       (reduce bdd/and (bdd/and bdd1 bdd2) bdds))))

398         (= true bdd1) true

504  (defn type-subtype?
505    "Given two type designators, use bdds to determine whether one is a subtype of the other.
506    if it cannot be proven, false is returned."
507    [subtype-designator supertype-designator]
508    (with-hash []
509      (let [bdd-sub (bdd subtype-designator)
510            bdd-sup (bdd supertype-designator)
511            bdd-diff (bdd/and-not bdd-sub bdd-sup)
512            satisfying (satisfying-type-designators bdd-diff)]
513        ;; There are four conditions to detect:
514        ;;  1. if the bdd is identically false
515        ;;     subtype -> true
516        ;;  2. if every branch in the bdd is unihabited i.e., inhabited -> false
517        ;;     then subtype -> true.   Note that 2 implies 1, so potentially
518        ;;     there may really be no need to check 1?? not sure
519        ;;  3. if some branch x exists for which inhabited(x :dont-know) = true
520        ;;     subtype -> false
521        ;;  4. thus there necessarily exists some branch x for which inhibited(x :dont-know) = :dont-know,
522        ;;     but no branch y for which inhabited(y :dont-know) = true
523        ;;     then subtype -> :dont-know.
524        ;;     Note that !2 and !3 => 4, so there's no need to check anything on 4.      
525        (cond (= false bdd-diff)
526              true
527              (forall [td satisfying]
528                      (or (= false (gns/inhabited? td :dont-know))
529                          (= false (gns/inhabited? (gns/canonicalize-type td) :dont-know))))
530              true
531  
532              (exists [td satisfying]
533                      (or (= true (gns/inhabited? td :dont-know))
534                          (= true (gns/inhabited? (gns/canonicalize-type td) :dont-know))))
535              false
536              
537              :else
538              (do (assert (exists [td satisfying]
539                                  (= :dont-know (gns/inhabited? td :dont-know))))
540                  ;; in the case the bdd is not explicitly false, some branch(s)
541                  ;;   from the root to the true-leaf might designate a list of
542                  ;;   types whose intersection is empty.
543                  ;;   The two given types are in a sub/super relation ONLY
544                  ;;   if all such intersections are empty.
545                  :dont-know)))))
