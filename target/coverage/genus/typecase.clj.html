027  (defn- collect-leaf-types [tds]
028    (letfn [(collect [td]
029              (cond (gns/not? td)
030                    (collect (second td))
031                    (gns/and? td)
032                    (mapcat collect (rest td))
033                    (gns/or? td)
034                    (mapcat collect (rest td))
035                    :else
036                    (list td))
037              )]
038      (mapcat collect tds)))

040  (defn- canonicalize-pairs
041    "pairs is a sequence of items which are implicitly (not explicitly grouped),
042    e.g., pairs = (a 100 b 200 c 300).  This function interprets "
043    [pairs]
044    
045    (if (empty? pairs)
046      ()
047      (let [[td consequent & others] pairs]
048        `([~(gns/canonicalize-type td) ~consequent] ~@(canonicalize-pairs others)))))

050  (defn- substitute-1-type
051    "Helper function for substitute-type, which does the search/replace
052    on one given type designators, td-domain.  This function search for
053    occurances of td-search in td-domain, and if found, replaces with
054    td-replace."
055    [td-search td-replace td-domain]
056    (cond (= td-search td-domain)
057          td-replace
058          (gns/not? td-domain)
059          (template (not ~(substitute-1-type td-search td-replace (second td-domain))))
060          (gns/and? td-domain)
061          (template (and ~@(map (fn [td] (substitute-1-type td-search td-replace td)) (rest td-domain))))
062          (gns/or? td-domain)
063          (template (or ~@(map (fn [td] (substitute-1-type td-search td-replace td)) (rest td-domain))))
064          :else
065          td-domain))

067  (defn- substitute-type
068    "pairs is a sequence of items each of the form [type-designator consequent].
069    This function attempts to simplify each of the type designators, return a
070    new sequence of the same length, with the consequents untouched.  Each
071    type-designator has had occurances of the type designator td-old replaced
072    with td-new."
073    [td-old td-new pairs]
074    
075    (for [[td consequent] pairs]
076      [(gns/canonicalize-type (substitute-1-type td-old td-new td))
077       consequent]))

079  (defn- substitute-types
080    "Helper function which performs multiple type designator substitutions."
081    [td-olds td-new pairs]
082    (loop [td-olds td-olds
083           pairs pairs]
084      (if (empty? td-olds)
085        pairs
086        (recur (rest td-olds)
087               (substitute-type (first td-olds) td-new pairs)))))

089  (defn- prune-pairs
090    "This is a helper function used in optimizing the macro expansion
091    of typecase.  pairs is a sequence of objects, each of the form 
092    [type-designator expression].  This function returns a new sequence
093    which prunes (removes the trailing elements) the sequence at the first
094    occurance of [:sigma ...], if such occurs, removes any pair
095    of the form [:empty-set ...]."
096    [pairs]
097    (cond (empty? pairs)
098          nil
099  
100          :else
101          (let [[[td consequent] & others] pairs]
102            (case td
103              (:sigma) (list [:sigma consequent])
104              (:empty-set) (prune-pairs others)
105              (cons [td consequent] (prune-pairs others))))))

107  (defn- calc-influence [all-freq all-leaves leaf]
108    (let [disjoints (setof [t all-leaves] (= true (gns/disjoint? t leaf :dont-know)))
109          proper-supers    (setof [t all-leaves] (and (not= t leaf)
110                                               (= true (gns/subtype? leaf t :dont-know))))
111          proper-subs      (setof [t all-leaves] (and (not= t leaf)
112                                                      (= true (gns/subtype? t leaf :dont-know))))
113          f (fn [sum leaf] (+ sum (get all-freq leaf)))
114          influence   (+ (get all-freq leaf)
115                         (reduce f 0 disjoints)
116                         (reduce f 0 proper-supers)
117                         (reduce f 0 proper-subs))]
118      [leaf influence disjoints proper-supers proper-subs]))
119  
120  (defn- maximize-influence [canonicalized-pairs]
121    (let [all-leaves (collect-leaf-types (map first canonicalized-pairs))
122          all-freq (frequencies all-leaves)
123          first-leaves (collect-leaf-types (list (first (first canonicalized-pairs))))
124          ]
125      ;; find the leaf in first-leaves with the maximum occurances counting disjoints, supers, and subs 
126      (reduce (fn [[most-freq' influence' disjoints' supers' subs'] leaf]
127                 (let [[most-freq influence disjoints supers subs] (calc-influence all-freq all-leaves leaf)]
128                   (if (> influence' influence)
129                     [most-freq' influence' disjoints' supers' subs']
130                     [most-freq  influence  disjoints  supers  subs])))
131               [:ignore 0 () () ()]
132               first-leaves)))

134  (defmacro optimized-let
135    "Macro used in the macro expansion of typecase.
136    This is a very simple let form which checks for a very special
137    case such as (let [x x] y) and optimizes it simply to y."
138    [[a b] c]
139    (if (= a b)
140      c
141      `(let [~a ~b] ~c)))

143  (defmacro optimized-typep
144    "Macro used to optimize the a call such as (typep x '(satisfies y))
145    to (y x)
146  
147    (optimized-typep 234 (not (satisfies y)))
148    --> (not (optimized-typep 234 (satisfies y)))
149  
150    (optimized-typep 234 (satisfies y))
151    --> (boolean (y 234))
152  
153    (optimized-typep 234 xyzzy)
154    --> (gns/typep 234 'xyzzy)"
155    [v t]
156    (cond (gns/satisfies? t)
157          (let [[_ pred] t]
158            `(boolean (~pred ~v)))
159  
160          (gns/not? t)
161          (let [[_not ty] t]
162          `(not (optimized-typep ~v ~ty)))
163  
164          (and (symbol? t)
165               (gns/safe-resolve t) ;; only flatten class names which safely resolve
166               ;;  i.e., Number but not Ratio
167               (gns/resolve-class t))
168          `(instance? ~t ~v)
169  
170          :otherwise
171          `(gns/typep ~v '~t)))

174  (defmacro typecase 
175    "Takes an expression and a set of clauses
176    of the form ... type-designator consequent ...
177    and an optional final default value.
178  
179    The evaluation semantics are the following:
180    The first (type-designator consequent) value is chosen
181    for which the given value is of that type (as per genus/typep),
182    and the consequent is evaluated.  If multiple type-designators
183    match the value, only the first consequent is evaluated.
184  
185    E.g.,
186    (typecase 42
187      String (println 100)
188      (or String (satisfies oddp)) (println 200)
189      (and Integer (satisfies evenp)) (println 300) ;; <-- this is chosen and evaluated
190      (or String Integer) (println 400)
191      (println 500))
192  
193    The type-designators are those as specified in genus.
194    The given expression is evaluated a maximum of once, but in some
195    optimized cases it may not be evaluated at all.  If none of the type
196    designators designate a type for which the given value is a member,
197    then either the default value is returned, or nil if no default is
198    given.
199    Otherwise the consequent is evaluated and returned corresponding
200    to the first (top most) type designator which matches the value.
201    These type checks are highly optimized so that no type explicitly
202    mentioned is checked more than once, including (and especially)
203    (satisfies ...)."
204    [value & pairs]
205    (if (odd? (count pairs))
206      `(typecase ~value ~@(butlast pairs) :sigma ~(last pairs))
207          
208      (let [canonicalized-pairs (prune-pairs (canonicalize-pairs pairs))]
209        (if (empty? canonicalized-pairs)
210          nil
211          (let [[most-freq influence disjoints proper-supers proper-subs]
212                (maximize-influence canonicalized-pairs)]
213            (cond
214              (and (= 1 (count canonicalized-pairs))
215                   (= :sigma (first (first canonicalized-pairs))))
216              (second (first canonicalized-pairs))
217              
218              (= 1 influence)
219              (let [[[type consequent] & others] canonicalized-pairs]
220                ;; canonicalized-pairs ([type consequent] [type consequent] ...)
221                (if (gns/not? type)
222                  (if (symbol? value)
223                    ;; if the type is (not x), then strip off the not, and swap the then/else
224                    `(if (optimized-typep ~value ~(second type))
225                       (typecase ~value ~@(mapcat identity others))
226                       ~consequent)
227                    `(optimized-let [value# ~value]
228                                    ;; if the type is (not x), then strip off the not, and swap the then/else
229                                    (if (optimized-typep value# ~(second type))
230                                      (typecase value# ~@(mapcat identity others))
231                                      ~consequent)))
232                  (if (symbol? value)
233                    `(if (optimized-typep ~value ~type)
234                       ~consequent
235                       (typecase ~value ~@(mapcat identity others)))
236                    
237                    `(optimized-let [value# ~value]
238                                    (if (optimized-typep value# ~type)
239                                      ~consequent
240                                      (typecase value# ~@(mapcat identity others)))))))
241  
242              :else
243              ;; most-freq is a leaf-level type which appears multiple times, either
244              ;; appearing explicitly, or disjoint, sub/super types also appear.
245              ;; we expand to two cases, one where the type is matched, and one where
246              ;; the complement of the type is not matched.
247              (let [if-true  (substitute-types disjoints :empty-set
248                                               (substitute-types (cons most-freq proper-supers)
249                                                                 :sigma canonicalized-pairs))
250                    if-false (substitute-types (cons most-freq proper-subs)
251                                               :empty-set canonicalized-pairs)]
252                (if (symbol? value)
253                  `(if (optimized-typep ~value ~most-freq)
254                     (typecase ~value ~@(mapcat identity if-true))
255                     (typecase ~value ~@(mapcat identity if-false)))
256                  `(optimized-let [value# ~value]
257                                  (if (optimized-typep value# ~most-freq)
258                                    (typecase value# ~@(mapcat identity if-true))
259                                    (typecase value# ~@(mapcat identity if-false))))))))))))
