033  (def gs/spec?
034    "Detect sequence starting with the simple symbol spec"
035    (seq-matcher 'spec))

037  (defmethod gns/typep 'spec [a-value [_a-type pattern]]
038    (assert (not= nil pattern) "gns/typep: nil is not a supported spec")
039    (try (boolean (s/valid? pattern a-value))
040         (catch Exception e
041           (cl-format true "s/valid? failed on a-value=~A pattern=~A"
042                      a-value pattern)
043           (throw e))))
044  
045  (defmethod gns/valid-type? 'spec [[_ pattern]]
046    (boolean (s/get-spec pattern)))
047  
048  (defmethod gns/-inhabited? 'spec [_t1]
049    :dont-know)
050  
051  (defmethod gns/-disjoint? 'spec [t1 t2]
052    (cond (not (gs/spec? t1))
053          :dont-know
054  
055          (and (= t1 t2)
056               (gns/inhabited? t1 false))
057          false
058  
059          :else ;; for the moment assume that spec is always disjoint with anything
060          false))
061  
062  (defmethod gns/-subtype? 'spec [_sub-designator _super-designator]
063    :dont-know)

065  (defn symbol= [x y]
066    (let [ns (find-ns 'genus.genus-spec)]
067      (assert ns "cannot find name space")
068      (= (ns-resolve ns x)
069         (ns-resolve ns y))))

071  (defn recuperate-closure
072    "Takes an expression something like:
073    (clojure.core/fn [%] (clojure.core/> % 1000))
074    and returns an equivalent function object.
075    In case the function object has a free variable, it
076    is not possible to recuperate a function object, so
077    an Exception is thrown"
078    [expr]
079    {:pre [(and (sequential? expr)
080                (symbol= (first expr) 'fn)
081                (vector? (second expr)))]}
082    (eval expr))

084  (defn wrap-spec 
085    [pattern]
086    (cond (not (sequential? pattern))
087          pattern
088  
089          (empty? pattern)
090          pattern
091          
092          (exists [k1 '(s/cat s/alt s/* s/? s/+)]
093                  (symbol= k1 (first pattern)))
094          (list 'spec pattern)
095  
096          :else
097          pattern))

099  (def spec-to-rte
100    "Convert spec to rte"
101    (gc-friendly-memoize
102      (fn [pattern]
103    (cond (s/regex? pattern)
104          (spec-to-rte (s/form pattern))
105  
106          (s/spec? pattern)
107          (spec-to-rte (s/form pattern))
108  
109          (s/get-spec pattern)
110          (spec-to-rte (s/form pattern))
111      
112          (and (symbol? pattern)
113               (resolve pattern)
114               (fn? (deref (resolve pattern))))
115          (list 'satisfies pattern)
116  
117          (sequential? pattern)
118          (letfn [(specify [operands]
119                    (for [operand operands]
120                      (list 'spec operand)))
121                  (transform-sequence-pattern [pattern]
122                    (if (sequential? pattern)
123                      (let [[tag & operands] pattern]
124                        (casep symbol= tag
125                          (s/cat) (cons :cat (transform-sequence-patterns (strip-keys operands)))
126                          (s/alt) (cons :or (transform-sequence-patterns (strip-keys operands)))
127                          (s/*)   (cons :* (transform-sequence-patterns operands))
128                          (s/?)   (cons :? (transform-sequence-patterns operands))
129                          (s/+)   (cons :+ (transform-sequence-patterns operands))
130                          (s/&)   (unsupported pattern)
131                          (list 'spec pattern)))
132                      (list 'spec pattern)))
133                  (transform-sequence-patterns [patterns]
134                    (map transform-sequence-pattern patterns))
135                  (strip-keys [operands]
136                    (assert (sequential? operands)
137                            (cl-format false "expecting a sequence, not ~A" operands))
138                    (let [stripped
139                          (loop [operands (doall operands)
140                                 stripped ()]
141                            (cond (empty? operands)
142                                  (reverse stripped)
143  
144                                  :else
145                                  (do
146                                    (when (not (keyword? (first operands)))
147                                      (cl-format true "WARNING: expecting keyword, not ~A in ~A"
148                                                 (first operands) operands))
149                                    (recur (rest (rest operands))
150                                           (cons (second operands)  stripped)))))]
151                      stripped))
152                  (unsupported [pattern]
153                    (throw (ex-info "unsupported pattern" {:unsupported-pattern true
154                                                           :pattern pattern
155                                                           :thrown-by 'spec-to-rte})))]
156            (let [[tag & operands] pattern
157                  transformed (casep symbol= tag
158                                     (fn) (template (satisfies ~(recuperate-closure pattern)))
159                                     (s/and) (template (and ~@(specify operands)))
160                                     (s/or) (template (or ~@(specify (strip-keys operands))))
161                                     (s/nilable) (template (or (satisfies nil?) ~@(specify operands)))
162                                     (s/keys
163                                      s/keys*
164                                      s/coll-of) (template (spec ~(eval pattern)))
165                                     
166                                     (s/cat
167                                      s/alt
168                                      s/*
169                                      s/?
170                                      s/+) (template (rte ~(transform-sequence-pattern pattern)))
171                                     (s/&) (unsupported pattern)
172                                     
173                                     pattern)
174                  ]
175              transformed 
176              ))
177  
178          :else
179          (list 'spec pattern)
180              ))))

182  (defmethod gns/-canonicalize-type 'spec method-canonicalize-type-spec
183    [[_spec pattern] _nf]
184    (try (wrap-spec (spec-to-rte pattern))
185         (catch clojure.lang.ExceptionInfo ei
186           (if (:unsupported-pattern (ex-data ei))
187             (do ;; if we fail to expand the pattern, then don't even try
188               (cl-format true "failed to canonicalize type: ~A, at ~A~%"
189                          pattern (:pattern (ex-data ei)))
190               (list 'spec pattern))
191             (throw ei)))))
