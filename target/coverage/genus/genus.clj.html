264  (defn extract-type-from-expression
265    "After the expression representing the code body has been extracted from the code body
266    of a type predicate, use several heursitics to match the code, to extract the type
267    which is being implemented in the expression.
268    Return nil, if the match is not successful.
269    Otherwise return a type-designator.
270    Examples:
271      (extract-type-from-expression 'n '(instance? BigDecimal n))
272        ==> BigDecimal
273      (extract-type-from-expression 'n '(or (integer? n) (ratio? n) (decimal? n)))
274        ==> (or (or Integer Long clojure.lang.BigInt BigInteger Short Byte)
275                clojure.lang.Ratio BigDecimal)
276      ;; if variable 'X is different from variable used inside expression, return nil
277      (extract-type-from-expression 'X '(instance? BigDecimal n))
278       ==> nil
279    "
280    [var-1 expr]
281    (cond
282      (not (sequential? expr))
283      nil
284      
285      ;; (instance? clojure.lang.Symbol x)
286      (and (= 3 (count expr))
287           (= 'instance? (first expr)))
288      (let [[_i type-designator var-2] expr]
289        (if (and (= var-1 var-2)
290                 (symbol? type-designator))
291          type-designator
292          nil))
293      
294      ;; (symbol? x)
295      (= 2 (count expr))
296      (let [[type-predicate var-2] expr]
297        (if (and (= var-1 var-2)
298                 (symbol? type-predicate))
299          (type-predicate-to-type-designator type-predicate)
300          nil))
301  
302      ;; (or ...)
303      (= 'or (first expr))
304      (let  [[_or & exprs] expr
305             expanded (map (fn [ex]
306                             (extract-type-from-expression var-1 ex))
307                           exprs)
308             ]
309        (if (not (member nil expanded))
310          (cons 'or expanded)
311          nil))
312  
313      :else
314      nil))

316  (def type-predicate-to-type-designator 
317    "Look at the function definition s-expression of the named function, type-predicate,
318    and apply heuristics to attempt to reverse-engineer the type being checked.  
319    This works for type predicates as they are defined in core.clj.
320    Examples:
321    (type-predicate-to-type-designator 'int?)
322       ==>  '(or Long Integer Short Byte)
323    (type-predicate-to-type-designator 'rational?)
324      ==> (or (or Integer Long clojure.lang.BigInt BigInteger Short Byte)
325              clojure.lang.Ratio BigDecimal)
326    (type-predicate-to-type-designator 'ident?)
327    ==> (or clojure.lang.Keyword clojure.lang.Symbol)"
328    (fn [type-predicate]
329      (let [fn-s-expression (get-fn-source type-predicate)]
330        (cond
331          (not (sequential? fn-s-expression))
332          nil
333  
334          (empty? fn-s-expression)
335          nil
336  
337          (= 3 (count fn-s-expression))
338          (try (let [[_def name-1 [_fn name-2 [var-1] expr]] fn-s-expression]
339                 (if (and (= name-1 name-2)
340                          (= _def 'def)
341                          (= _fn 'fn))
342                   (extract-type-from-expression var-1 expr)
343                   nil))
344               (catch Exception _
345                 nil))
346          
347          (= 6 (count fn-s-expression))
348          (try
349            (let [[_defn name doc-string attr-map [var-1] expr]
350                  fn-s-expression]
351              (if (and (= _defn 'defn)
352                       (symbol? name)
353                       (string? doc-string)
354                       (map? attr-map)
355                       (symbol? var-1))
356                (extract-type-from-expression var-1 expr)
357                nil))
358            (catch Exception _
359              nil))
360  
361          :else
362          nil))))

423    :empty-set handles the decision of (typep 3 :empty-set)."

485  ;; if it is = to x (according to the clojure = function)
486  ;; and also it has the same type.   Thus a lazy list is not equal to a list

493  ;; if v == x or v == y or v == z (according to the clojure = function)
