027  (defn safe-odd? 
028    "Save version of odd?.  The build-in odd? throws
029    an exception if its argument is not an integer.
030    safe-odd? simply returns false in this case."
031    [n]
032    (and (integer? n)
033         (odd? n)))
034  
035  (defn safe-positive? [n]
036    "Save version of positive?.  The build-in positive? throws
037    an exception if its argument is not an number.
038    safe-positive? simply returns false in this case."
039    (and (number? n)
040         (> n 0)))

097  (defn gen-type
098    "Generate a type designator for testing."
099    ([size]
100     (gen-type size *test-types* false))
101    ([size types]
102     (gen-type size types false))
103    ([size types limit-to-inhabited]
104     (if (< 0 size)
105       (case (rand-nth '(or and not :else))
106         (or) (template (or ~(gen-type (dec size) types limit-to-inhabited)
107                            ~(gen-type (dec size) types limit-to-inhabited)))
108         (and) (let [t1 (if limit-to-inhabited
109                          (gen-inhabited-type (dec size) types)
110                          (gen-type (dec size) types limit-to-inhabited))
111                     t2 (if limit-to-inhabited
112                          (gen-inhabited-type (dec size) types)
113                          (gen-type (dec size) types limit-to-inhabited))]
114                 ;; when trying to generate an inhabited type, and in the AND case
115                 ;;   we must make sure that both operands of (and ...) designate
116                 ;;   inhabited types.  If not, we can correct this now, rather than
117                 ;;   waiting to the top level
118                 (template (and ~t1 ~t2)))
119         (not) (template (not ~(gen-type (dec size) types)))
120         (rand-nth types))
121       (rand-nth types))))

123  (defn tree-012-to-type [tree types]
124    (case (count tree)
125      (0) (rand-nth types)
126      (2) (let [[parent child] tree]
127            (gns/Not (tree-012-to-type child types)))
128      (3) (let [[parent left right] tree]
129            (list (rand-nth '(or and))
130                  (tree-012-to-type left types)
131                  (tree-012-to-type right types)))))

133  (defn gen-balanced-type
134    "Generate a type designator for testing."
135    ([depth]
136     (gen-balanced-type depth *test-types*))
137    ([depth types]
138     (if (< 0 depth)
139       (tree-012-to-type (rand-tree-012 0.66666 depth) types)
140       (rand-nth types))))

142  (defn gen-inhabited-type 
143    "gen-type may generate a type designator which reduces to :empty-set.
144    gen-inhabited-type generates a type designator which is guaranteed
145    be inhabited, i.e. gns/inhabited? returns true"
146    ([size]
147     (gen-inhabited-type size *test-types*))
148    ([size types]
149     (loop []
150       (let [td (gen-balanced-type size types)]
151         (case (gns/inhabited? td :dont-know)
152           (true)
153           td
154  
155           (false :dont-know)
156           (recur))))))

158  (defn gen-quasi-inhabited-type 
159    "gen-type may generate a type designator which reduces to :empty-set.
160    gen-inhabited-type generates a type designator which not provably empty.
161    I.e., gns/inhabited? returns true or :dont-know"
162    ([size]
163     (gen-quasi-inhabited-type size *test-types*))
164    ([size types]
165     (loop []
166       (let [td (gen-balanced-type size types) ]
167         (case (gns/inhabited? td :dont-know)
168           (true :dont-know)
169           td
170  
171           (false)
172           (recur))))))
