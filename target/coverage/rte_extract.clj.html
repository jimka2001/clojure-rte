028  (defn extract-rte
029    "Accepts an object of type Dfa, and returns a map which associates
030    exit values of the dfa with canonicalized rte patterns of the accepting
031    langauge. If there are no accepting states in the Dfa, an empty map {}
032    is returned."
033    [dfa']
034    ;; TODO - this can be done easier
035    ;;    1. minimize and trim the given dfa
036    ;;    2. generate a list of transition triples [from label to]
037    ;;    3. add transitions from extra-state-I to all initial states with :epsilon transition
038    ;;    4. add transitions from all accepting states to extra-state-F (one per exit value) with :epsilon transition
039    ;;    5. loop on each state
040    ;;    6.    partition transitions into 4 groups [to-this-state loops-on-state from-state everything-else]
041    ;;    7.    combine parallel transitions
042    ;;    8.    n^2 iteration to-this-state x from-this-state
043    ;;    9.    append new transitions in next iteration of loop 5.
044    ;;    10. this reduces to one transition per exit value, returns the map of exit-value to label
045    ;; assert that state 0 is the initial state
046    ;; and that no other state is an initial state
047    (doseq [[id state] (:states dfa')]
048      (if (= id 0)
049        (assert (:initial state) (format "%s should be an initial state" state))
050        (assert (not (:initial state)) (format "%s should NOT be an initial state" state))))
051    (let [;; #1
052          dfa (xym/trim (xym/minimize dfa')) ; we must minimize otherwise the size of the returned expression can be HUGE
053          ;; #2
054          old-transition-triples (for [q (xym/states-as-seq dfa)
055                                       [label dst-id] (:transitions q)]
056                                   [(:index q) label dst-id])
057          ;; #3
058          new-initial-transitions [[:I :epsilon 0]]
059          ;; #4
060          new-final-transitions (for [q (xym/states-as-seq dfa)
061                                      :when (:accepting q)]
062                                  ;; we designate new final states each as [:F some-exit-value]
063                                  [(:index q) :epsilon [:F (xym/exit-value dfa q)]])
064          ]
065      (doseq [[id state] (:states dfa)]
066        (if (= id 0)
067          (assert (:initial state) (format "%s should be an initial state" state))
068          (assert (not (:initial state)) (format "%s should NOT be an initial state" state))))
069  
070      (letfn [;; local function
071              (combine-parallel-labels [operands]
072                (rte/canonicalize-pattern
073                 (rte/create-or operands)))
074  
075              ;; local function
076              (extract-labels [triples]
077                (map second triples))
078  
079              ;; local function
080              (combine-parallel [triples]
081                ;; accepts a sequence of triples, each of the form [from label to]
082                ;;   groups them by common from/to, these are parallel transitions
083                ;;   combines the labels of the parallel transitions, into one single label
084                ;;   and collects a sequence of transitions, none of which are parallel.
085                ;;   This action is important because it greatly reduces the number of transitions
086                ;;   created.  The caller, the computation of new-triples, makes an NxM loop
087                ;;   creating NxM new triples.   This reduces N and M by eliminating parallel
088                ;;   transitions.
089                (for [[[from to] triples] (group-by (fn [[from _ to]] [from to]) triples)
090                      :let [label (combine-parallel-labels (extract-labels triples))]
091                      ]
092                  [from label to]))
093  
094              ;; local function
095              (eliminate-state [transition-triples q-id]
096                (when (timeout-reached?)
097                  (throw (ex-info (format "extract-rte interrupted while eliminating state %s" q-id)
098                                  {:remaining-triples (count transition-triples)})))
099                (let [[x-to-q q-to-q q-to-x others]
100                      ;; #6
101                      (reduce (fn [[x-to-q q-to-q q-to-x others]
102                                   [src-id _label dst-id :as triple]]
103                                (cond
104                                  (and (= src-id q-id)
105                                       (= dst-id q-id))
106                                  ;; extend q-to-q
107                                  [x-to-q (cons triple q-to-q) q-to-x others]
108  
109                                  (= src-id q-id)
110                                  ;; extend q-to-x
111                                  [x-to-q q-to-q (cons triple q-to-x) others]
112                                  
113                                  (= dst-id q-id)
114                                  ;; extend x-to-q
115                                  [(cons triple x-to-q) q-to-q q-to-x others]
116  
117                                  :else
118                                  ;; extend others
119                                  [x-to-q q-to-q q-to-x (conj others triple)]))
120                              [() () () []]
121                              transition-triples)
122  
123                      ;; #7
124                      self-loop-label (combine-parallel-labels (extract-labels q-to-q))
125                      ;; #8
126                      new-triples (for [[src pre-label _] (combine-parallel x-to-q)
127                                        [_ post-label dst] (combine-parallel q-to-x)]
128                                    [src
129                                     (rte/create-cat (list pre-label
130                                                           (list :* self-loop-label)
131                                                           post-label))
132                                     dst])]
133  
134                  (concat others new-triples)))]
135  
136        ;; #5 / #9
137        (let [new-transition-triples (reduce eliminate-state
138                                             (concat new-initial-transitions
139                                                     old-transition-triples
140                                                     new-final-transitions)
141                                             ;; TODO need to order states for faster
142                                             ;;    elimination.  I.e., order by increasing
143                                             ;;    product of num-inputs x num-outputs
144                                             (xym/ids-as-seq dfa))
145              grouped (group-by (fn [[_ _ [_ exit-value]]] exit-value) new-transition-triples)]
146          (into {}
147                (doall 
148                 (for [[exit-value triples] grouped
149                       :let [pretty (combine-parallel-labels (extract-labels triples))]
150                       ]
151                   ;; one label per return value
152                   ;; #10
153                   [exit-value pretty])))))))

155  (defn dfa-to-rte
156    "Accepts an object of type Dfa, and returns a map which associates
157    exit values of the dfa with canonicalized rte patterns of the accepting
158    langauge.  If there are no accepting states in the Dfa, a map {true :empty-set}
159    is returned."
160    [dfa]
161    (assert (instance? (xym/record-name) dfa)
162            (cl-format false "dfa-to-rte: expecting Dfa, not ~A ~A" (type dfa) dfa))
163    (let [ret-val-rte (extract-rte dfa)]
164      (if (= {} ret-val-rte)
165        {true :empty-set}
166        ret-val-rte)))

168  (defn rte-equivalent?
169    "Compare two rte patterns by comparing their constructed Dfa's."
170    [rte-1 rte-2]
171    (xym/dfa-equivalent? (rte/rte-to-dfa rte-1)
172                         (rte/rte-to-dfa rte-2)))
