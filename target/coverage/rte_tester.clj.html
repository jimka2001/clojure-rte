036  (defn rte-components [pattern]
037    (cond
038      (and (seq? pattern)
039           (empty? pattern))
040      ()
041  
042      (seq? pattern)
043      (let [[keyword & operands] pattern]
044        (case keyword
045          (:* :+ :? :not
046              :and :or :cat :permute) operands
047          ;; case else
048          ()))
049  
050      :else
051      ()))

055  (defn rte-depth
056    "compute the depth of an RTE expression"
057    [r]
058  
059    (cond (not (sequential? r))
060          0
061  
062          (keyword? (first r))
063          (inc (reduce max 0 (map rte-depth (rest r))))
064  
065          :else
066          0))
067  
068  (defn rte-count-leaves 
069    "Count the leaf nodes of a given RTE"
070    [r]
071    (cond (not (sequential? r))
072          1
073  
074          (keyword? (first r))
075          (reduce + 0 (map rte-count-leaves (rest r)))
076  
077          :else
078          1))
