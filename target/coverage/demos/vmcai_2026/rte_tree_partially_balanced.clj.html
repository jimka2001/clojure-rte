009  (defn insert-tree 
010    "nondestructively insert a value into a binary tree.  nodes are of the form [value left right],
011    leaves are all empty according to the empty? function"
012    [value tree]
013    (cond (empty? tree)
014          [value nil nil]
015  
016          :else
017          (let [[top left right] tree]
018            (cond (< value top)
019                  [top
020                   (insert-tree value left)
021                   right]
022  
023                  (> value top)
024                  [top
025                   left
026                   (insert-tree value right)]
027  
028                  :else
029                  tree))))

032  (defn build-binary-tree
033    "Build a binary tree by iterating through the
034    population, and inserting them one by one in the given order. "
035    [population]
036    (loop [tree nil
037           population population]
038      (if (not-empty population)
039        (recur (insert-tree (first population) tree)
040               (rest population))
041        tree)))

046  (defn tree-to-rte 
047    "Take a tree as created by `build-binary-tree`, insert a type designator
048    (from the given `types`)  at each leaf node, and insert a randomly selected
049    operator at each internal node.
050    If the operator is unary, then the right child is simply discarded."
051    ([tree]
052     (tree-to-rte tree *test-types*))
053    ([tree types]
054     (cond (not-empty tree)
055           (let [key (rand-nth '[:? :* :not
056                                 :and :or :cat])
057                 [value left right] tree]
058             (case key
059               (:? :* :not) (if (= 0 (rand-int 2)) ;; 50% choice
060                              (list key (tree-to-rte left types))
061                              (list key (tree-to-rte tree types)))
062               (:and :or :cat) (list key
063                                     (tree-to-rte left types)
064                                     (tree-to-rte right types))))
065  
066           :else 
067           (rand-nth types))))

070  (defn rand-tree 
071    "Build a binary tree which has between 2^n and (2^(n+1) - 1) leaf nodes.
072    The tree is built by taking a random permutation of 
073    and inserting each of (range 2^n) a tree in random order
074    (starting with an empty tree)"
075  
076    [depth]
077    (let [num-leaves (+ (round (pow 2 depth))
078                        (rand-int (round (pow 2 depth))))
079          population (into () (shuffle (range num-leaves)))
080          tree (build-binary-tree population)
081          ]
082      ;; (printf "target num-leaves=%d %s\n" num-leaves tree)
083      tree
084      ))

088  (defn gen-partially-balanced-rte
089    "Generate an RTE which corresponds (on average) in shape closely to a balanced
090    binary tree.  The goal is to sample languages uniformly rather than sampling
091    syntactic structure of the RTE uniformly."
092    ([depth]
093     (gen-partially-balanced-rte depth 
094                       (cons :epsilon *test-types*)))
095    ([depth types]
096     ;;(printf "generating tree of depth %d:  %s <= count < %s\n" depth (pow 2 depth) (pow 2 (inc depth)))
097     ;; a binary tree of depth=n has 2^n <= m < 2^(n+1) leaves
098     ;; so generate a random number 2^n <= rand < 2^(n+1)
099     ;; i.e 2^n + (rand-int 2^(n+1) - 2^n)
100     ;;    2^n + (rand-int 2^n)
101     (let [tree (rand-tree depth)
102           rte (tree-to-rte tree types)]
103  
104       rte)))
