117  (defn write-inhabited-subset-stats-csv 
118    "Append statistics to the file `subset-csv`  resources/statistics/dfa-subset.csv
119    and to `inhabited-csv` resources/statistics/dfa-inhabited.csv.
120    The statistics registered in `subset-csv` specify information about subset-ness and disjoint-ness
121    of two randomly selected dfas.
122    The statistics registered in `inhabited-csv` specify information about inhabited-ness of
123    the language of the dfa.
124    "
125    [& {:keys [num-states num-transitions
126               type-size probability-indeterminate
127               timeout-sec]
128        :or {num-states 10
129             num-transitions 30
130             type-size 2
131             probability-indeterminate 0.15
132             timeout-sec (* 60 2) ;; 2 minutes
133             }}]
134    (let [;;csv-file-name subset-csv
135          exit-value true
136          ]
137      (doseq [dfa-1 (with-timeout timeout-sec []
138                      ;; sequence of Dfas for doseq iteration
139                      [(xym/minimize (gen-dfa :num-states num-states
140                                              :num-transitions num-transitions
141                                              :exit-value exit-value
142                                              :type-size type-size
143                                              :probability-indeterminate probability-indeterminate))])
144              dfa-2 (with-timeout timeout-sec []
145                      ;; sequence of Dfas for doseq iteration
146                      [(xym/minimize (gen-dfa :num-states num-states
147                                              :num-transitions num-transitions
148                                              :exit-value exit-value
149                                              :type-size type-size
150                                              :probability-indeterminate probability-indeterminate))])
151              dfa-and (with-timeout timeout-sec []
152                        ;; sequence of Dfas for doseq iteration
153                        [(xym/synchronized-intersection dfa-1 dfa-2)])
154              dfa-and-not (with-timeout timeout-sec []
155                            ;; sequence of Dfas for doseq iteration
156                            [(xym/synchronized-and-not dfa-1 dfa-2)])
157              dfa-empty-word (with-timeout timeout-sec []
158                               ;; sequence of Dfas for doseq iteration
159                               [(rte-to-dfa :epsilon exit-value)])
160              dfa-and-non-trivial (with-timeout timeout-sec []
161                                    [(xym/synchronized-and-not dfa-and dfa-empty-word)])
162              
163              :let [subset (xym/dfa-subset? dfa-1 dfa-2)
164                    overlap (xym/dfa-inhabited? dfa-and)
165                    non-trivial-overlap (xym/dfa-inhabited? dfa-and-non-trivial)
166                    [satisfiability path] (get (xym/find-trace-map dfa-1) exit-value)]
167              ]
168  
169        (lock/merge-file inhabited-csv
170                         (fn [out-file]
171                           ;;(cl-format out-file "# num-states, num-transitions, type-size, probability-indeterminate, min-dfa-state-count, min-dfa-transitions-count, count indeterminate transitions, inhabited dfa language~%")
172                           ;; num-states , num-transitions , type-size , probability-indeterminate ,
173                           (cl-format out-file "~d,~d,~d,~f" num-states num-transitions type-size probability-indeterminate)
174  
175                           ;; min-dfa-state-count ,
176                           (cl-format out-file ",~d" (count (:states dfa-1)))
177  
178                           ;; min-dfa-transitions-count ,
179                           (cl-format out-file ",~d" (reduce + 0 (map (fn [st] (count (:transitions st)))
180                                                                      (xym/states-as-seq dfa-1))))
181  
182                           ;; count indeterminate transitions
183                           (cl-format out-file ",~d" (count (for [q (xym/states-as-seq dfa-1)
184                                                                  [td _] (:transitions q)
185                                                                  :when (= :dont-know (gns/inhabited? td :dont-know))]
186                                                              1)))
187  
188                           ;; inhabited dfa language?
189                           (cl-format out-file ",~a" satisfiability)
190  
191                           (cl-format out-file "~%")))
192  
193        (lock/merge-file subset-csv
194                         (fn [out-file]
195                           ;;(cl-format out-file "# num-states, num-transitions, type-size, probability-indeterminate, subset, overlap, non-trivial-overlap~%")
196                           
197                           ;; num-states , num-transitions , type-size , probability-indeterminate
198                           (cl-format out-file "~d,~d,~d,~f" num-states num-transitions type-size probability-indeterminate)
