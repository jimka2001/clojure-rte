050  (defn compute-balance-factors [r]
051    ;; {:post [(or (println [:r r :returns %]) true)]}
052    (cond (not (sequential? r))
053          [0 []]
054  
055          (keyword? (first r))
056          (let [[operator & operands] r
057                pairs (map compute-balance-factors operands)
058                heights (map first pairs)]
059            (if (empty? heights)
060              [0 []]
061              (let [max-height (reduce max heights)
062                    min-height (reduce min heights)
063                    balance (- max-height min-height) ]
064                [(inc max-height)
065                 (conj (mapcat second pairs) balance)])))
066  
067          :else
068          [0 []]))

073  (defn build-rtes
074    "Using a given genrator function `gen`, create `repetitions` many
075    rte objects.  Update the specified `csv-file` with data extracted
076    from the objects, one line per rte."
077    [depth repetitions & {:keys [gen csv-file verbose] ;; unary function which takes depth argument and generates a random rte of that approximate depth
078                          :or {gen gen-partially-balanced-rte
079                               csv-file gen-rte-partially-balanced-csv
080                               verbose false
081                               }}]
082    (let [time-out-secs 30
083          max-rte-leaf-count 35536
084          trivials '(:sigma (:* :sigma) (:cat :sigma (:* :sigma)) (:cat (:* :sigma) :sigma) :epsilon :empty-set)
085          freqs (loop [repetitions repetitions
086                       rtes nil]
087                  (if (not (pos? repetitions))
088                    (frequencies rtes)
089                    (truthy-let [rte-1 (gen depth)
090                                 count-rte-1-leaves (rte-count-leaves rte-1)
091                                 [_ balance-factors] (compute-balance-factors rte-1)
092                                 mean-balance (float (mean balance-factors 0))
093                                 max-balance (reduce max (conj balance-factors 0))
094                                 sigma-balance (std-deviation balance-factors 0)
095                                 :do (when (println [:rep repetitions
096                                                     :time (human-readable-current-time)
097                                                     :rte-1-count count-rte-1-leaves
098                                                     :mean-balance mean-balance
099                                                     :max-balance max-balance
100                                                     ;; :sigma-balance sigma-balance
101                                                     ]))
102                                 dfa-1 (when (> max-rte-leaf-count count-rte-1-leaves)
103                                         (with-timeout time-out-secs
104                                             (do (printf "timed out computing rte-to-dfa %s\n" rte-1)
105                                                 nil)
106                                           (do ;;(println "starting (rte-to-dfa rte-1)\n")
107                                             (rte-to-dfa rte-1))))
108                                 :do (when verbose (tab) (println [:dfa-1 dfa-1]))
109                                 dfa-min (with-timeout time-out-secs
110                                             (do (printf "timed out computing minimize %s\n" dfa-1)
111                                                 nil)
112                                           (xym/minimize dfa-1))
113                                 :do (when verbose (tab) (println [:dfa-min dfa-min]))
114                                 rte-map (with-timeout time-out-secs
115                                             (do (printf "timed out computing dfa-to-rte %s\n" dfa-min)
116                                                 nil)
117                                           (dfa-to-rte dfa-min))
118                                 rte-2 (when rte-map (get rte-map true :empty-set))
119                                 :do (if verbose
120                                       (do (tab) (println [:rte-2-count (rte-count-leaves rte-2)]))
121                                       (printf "skipped computation of rte-2\n"))
122  
123                                 dfa-2 (when (> max-rte-leaf-count (rte-count-leaves rte-2))
124                                         (with-timeout time-out-secs
125                                             (do (printf "timed out computing rte-to-dfa %s\n" rte-2)
126                                                 nil)
127                                           (rte-to-dfa rte-2)))]
128  
129                      ;; (println [:rep repetitions
130                      ;;           :depth depth
131                      ;;           :csv-file (cond (= csv-file gen-rte-partially-balanced-csv)
132                      ;;                           "partially-balanced"
133                      ;;                           (= csv-file gen-rte-totally-balanced-csv)
134                      ;;                           "totally-balanced"
135                      ;;                           (= csv-file gen-rte-classic-csv)
136                      ;;                           "classic")
137                      ;;           :date  (human-readable-current-time)])
138                      (if dfa-2
139                        (do (lock/merge-file csv-file
140                                             (fn [out-file]
141                                               ;;(cl-format out-file "# requested-depth, actual-depth, max-balance, mean-balance, sigma-balance, actual-leaf-count, minimized-depth, minimized-leaf-count, actual-state-count, minimized-state-count, minimized-rte~%")
142                                               (cl-format out-file "~D, ~D, ~D" depth (rte-depth rte-1) (rte-count-leaves rte-1))
143                                               (cl-format out-file ", ~A, ~A, ~A" max-balance mean-balance sigma-balance)
144                                               (cl-format out-file ", ~D, ~D" (rte-depth rte-2) (rte-count-leaves rte-2))
145                                               (cl-format out-file ", ~D, ~D" (count (xym/states-as-seq dfa-1)) (count (xym/states-as-seq dfa-2)))
146                                               (cl-format out-file ", ~A" (if (member rte-2 trivials)
147                                                                            rte-2
148                                                                            :other))
149                                               (cl-format out-file "~%")))
150                            (recur (dec repetitions) (cons rte-2 rtes)))
151                        (recur (dec repetitions) rtes)))))]
152      (into {} (for [[k v] freqs
153                     :when (> v 1)]
154                 [k v]))))
