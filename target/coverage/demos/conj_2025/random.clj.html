021  (defn tree-split-rte [algo leaves pivot]
022    {:pre [(fn? pivot) (int? leaves)]
023     :post [(rte/valid-rte? %)]}
024    (assert (< 0 leaves))
025    (if (= 1 leaves)
026      (rand-nth inhabited-leaves)
027      (let [rte (let [left-size (pivot leaves)]
028                  (assert (< 0 left-size))
029                  (letfn [(left  [] (tree-split-rte algo left-size pivot))
030                          (right [] (tree-split-rte algo (- leaves left-size) pivot))
031                          (mid   [] (tree-split-rte algo leaves pivot))]
032                    (weighted-case 20 (rte/Cat (left) (right))
033                                   30 (rte/And (left) (right))
034                                   30 (rte/Or (left) (right))
035                                   5  (rte/Star (mid))
036                                   15 (rte/Not (mid)))))
037            leaf-count (rte/count-leaves rte)]
038        ;; (println [:leaves :requested leaves :actual leaf-count])
039        rte)))
