012    (let [all-int '(:* (satisfies int?))
013          starts-0 '(:cat (= 0) (:* :sigma))
014          termine-0 '(:cat (:* :sigma) (= 0))
015          rte-even-parity '(:* (:cat :sigma :sigma))
016          rte-odd-parity (rte/Cat :sigma rte-even-parity)
017          r1 (rte/And starts-0
018                      rte-odd-parity)
019          r2 (rte/And termine-0
020                   rte-even-parity)
021          rte (rte/And all-int
022                       (rte/Or r1
023                               r2))
024          dfa (rte/rte-to-dfa rte)
025          ]
