037  (defn gen-csv-by-size [num-repetitions
038                         algo
039                         prefix
040                         lot
041                         min-leaf
042                         max-leaf]
043    (assert (gen-rte algo) (format "invalid algo=%s" algo))
044    (doseq [r (range num-repetitions)
045            :let [size (+ min-leaf (rand-int (- max-leaf min-leaf)))
046                  futures (for [q (range lot)]
047                            (future (do
048                                      #_(lock/with-lock
049                                        ;; using with-lock to prevent multiple printlns
050                                        ;; happing at the same time and writing a confusing log file
051                                        (println [:future :algo algo 
052                                                  :leaves size
053                                                  :rep (/ r num-repetitions)
054                                                  :iteration (/ q lot)]))                                      
055                                      (write-csv-statistic (fn [] ((gen-rte algo) size))
056                                                           algo
057                                                           (csv-file-name algo prefix)))))]]
058      (doall (map deref futures))))
