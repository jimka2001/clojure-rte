011  (defn plot-cb [base]
012    (assert (not (str/index-of base ".")))
013    (let [report-dir "doc/clojure.conj/2025/gnu/"
014          gnu-file-name (str report-dir base ".gnu")
015          png-file-name (str report-dir base ".png")
016          ]
017      (assert (.isDirectory (io/file report-dir)))
018      (fn [file-name-str]
019  
020        (println  (sh "cp" file-name-str gnu-file-name))
021        (println  (run-gnu-plot "png" gnu-file-name png-file-name)))))

025  (defn plot-histogram
026    ([] (plot-histogram "" true))
027    ([prefix view]
028     (let [buckets (for [algo algos
029                         :let [csv-lines (read-csv-lines algo prefix)
030                               state-counts (map :state-count csv-lines)]]
031                     [algo state-counts])]
032       (histogram buckets
033                  :base-name (if (= "" prefix)
034                               "histogram"
035                               (format "%s-histogram" prefix))
036  
037                  :x-label "DFA state count"
038                  :y-label "Percentage of DFAs per state count"
039                  :title (format "DFA State distribution for Rte %s" prefix)
040                  :gnu-file-CB (plot-cb (format "plot-%s-histogram" prefix))
041                  :keep-if #(<= % 9)
042                  :other-label ">= 10"
043                  :view view))))

045  (defn imb [cl]
046    (/ (:longest-branch cl)
047       (/ (log (:leaf-count cl))
048          (log 2))))

125  (defn gen-threshold-curve [csv-lines]
126    (let [num-samples (count csv-lines)
127          state-count-to-dfa-count (map (fn [[state-count lines-per-count]]
128                                          [state-count (count lines-per-count)])
129                                        (group-by :state-count csv-lines))
130          xys (for [[this-state-count _list] state-count-to-dfa-count
131                    :let [n (reduce + (for [[state-count num-lines] state-count-to-dfa-count
132                                            :when (>= state-count this-state-count)]
133                                        num-lines))]]
134                [this-state-count (* 100.0 (/ n num-samples))])]
135      (sort-by first xys)))
