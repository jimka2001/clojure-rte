014  (defn csv-file-name [algo prefix]
015    (str statistics-resource prefix algo ".csv"))

017  (defn write-csv-statistic [gen-rte prefix csv-file-name]
018    (let [rte (gen-rte)
019          leaf-count (rte/count-leaves rte)
020          node-count (+ leaf-count (rte/count-internal-nodes rte))
021          shortest (rte/measure-shortest-branch rte)
022          longest (rte/measure-longest-branch rte)
023          time-out-sec (max 60 (* 10 longest longest))
024          start (LocalDateTime/now)]
025      (letfn [(report [state-count transition-count min-state-count min-transition-count duration-ms]
026                (lock/merge-file csv-file-name
027                                 (fn [out-file]
028                                   (cl-format out-file "~D,~D," node-count leaf-count)
029                                   (doseq [s [state-count transition-count min-state-count min-transition-count]]
030                                     (if s
031                                       (cl-format out-file "~D," s)
032                                       (cl-format out-file "-1,")))
033                                   (cl-format out-file "~D,~D,~D~%" shortest longest duration-ms))))]
034        (doseq [dfa (with-timeout time-out-sec
035                        (do (printf "timed out computing rte-to-dfa %s\n" rte)
036                            (report nil nil nil nil time-out-sec)
037                            nil)
038                      [(rte/rte-to-dfa rte)])
039                :let [
040                      state-count (count (:states dfa))
041                      transition-count (reduce + 0 (map count (map :transitions (xym/states-as-seq dfa))))
042                      ]
043                min-dfa (with-timeout time-out-sec
044                            (do (printf "canceling DFA minimization after %d sec state-count=%d transition-count=%d"
045                                        time-out-sec state-count transition-count)
046                                (report state-count transition-count nil nil time-out-sec))
047                          [(xym/minimize dfa)])
048                :let [min-state-count (count (xym/states-as-seq min-dfa))
049                      min-transition-count (reduce + 0 (map count (map :transitions (xym/states-as-seq dfa))))
050                      end (LocalDateTime/now)
051                      duration (.toMillis (Duration/between start end))]
052                ]
053          (report state-count transition-count min-state-count min-transition-count duration)))))

056  (defn read-csv-lines [algo prefix]
057    (let [csv-file-name (csv-file-name algo prefix)
058          keys [:node-count
059                :leaf-count
060                :state-count
061                :transition-count
062                :minimized-state-count
063                :minimized-transition-count
064                :shortest-branch
065                :longest-branch
066                :duration-ms]
067          ]
068      (with-open [reader (io/reader csv-file-name)]
069        (assert reader (format "cannot read from %s" csv-file-name))
070        (doall
071         (for [csv-strings (csv/read-csv reader)
072               :when (not-empty csv-strings)
073               :when (not= \# (first (first csv-strings)))
074               :let [_ (assert (= (count keys) (count csv-strings))
075                               (format "invalid line %s in %s" csv-strings csv-file-name))
076                     csv-line (map Integer/parseInt csv-strings)
077                     parsed (into {} (map-indexed (fn [i k] [k (nth csv-line i)])
078                                                  keys))]]
079           parsed)))))
