004  (defn series-format-plot-data 
005    "encode plotting data into an hashmap ready to pass to oz/view!"
006    [chart-title x-label y-label data
007     & {:keys [sort x-scale y-scale]
008        :or {sort nil ;; or a vector of the exact order ["plot1" "plot2" "plot3"]
009             x-scale "linear" ;; or "log", "symlog", "sqrt" etc.
010             y-scale "linear"}}
011     ]
012    ;; data is of form [[string [(x y) (x y) (x y) ...]]
013    ;;                  [string [(x y) (x y) (x y) ...]]
014    ;;                  ...]
015  
016    (let [polished-data (for [[label aseq] data
017                              [x y] aseq]
018                          {x-label (float x)
019                           y-label (float y)
020                           :series label})]
021      {:data {:values polished-data}
022       :width 600
023       :height 600
024       :title {:text chart-title}
025       :description chart-title
026       :axes [{:orient "bottom"
027               :title x-label}
028              {:orient "left"
029               :title y-label}]
030       :encoding {:x {:field x-label
031                      :scale {:type x-scale}
032                      :type "quantitative"}
033                  :y {:field y-label
034                      :scale {:type y-scale}
035                      :type "quantitative"}
036                  :shape {:field "series"
037                          :sort sort
038                          :type "nominal"
039                          :legend {:orient "bottom"
040                                   :title "Curve shape"
041                                   :direction "vertical"
042                                   :labelLimit 1000}
043                          }
044                  :color {:field "series"
045                          :sort sort
046                          :legend {:orient "bottom"
047                                   :title "Curve color"
048                                   :direction "vertical"
049                                   :labelLimit 1000}
050                          :type "nominal"}}
051       :mark {:type "line" :point {:filled true :size 80}}}))
052  
053  (defn series-scatter-plot
054    "Pop up image in browser showing a graph with multiple curves.
055    Each curve corresponds to a series specified in the given `data`."
056    [chart-title x-label y-label data
057     & {:keys [x-scale y-scale sort]
058        :or {x-scale "linear" ;; or "log", "symlog", "sqrt" etc.
059             y-scale "linear"
060             sort nil
061             }}]
062    ;; data is of form [[string [(x y) (x y) (x y) ...]]
063    ;;                  [string [(x y) (x y) (x y) ...]]
064    ;;                  ...]
065    (let [tmp (.getAbsolutePath (java.io.File/createTempFile chart-title ".svg"))
066          formatted-data (series-format-plot-data chart-title x-label y-label data
067                                                  :sort sort
068                                                  :x-scale x-scale
069                                                  :y-scale y-scale)]
070  
071      (oz/export! formatted-data tmp)
072      tmp))
