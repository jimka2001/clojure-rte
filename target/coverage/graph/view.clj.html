001  (ns graph.view
002    (:require [clojure.java.shell :refer [sh]]
003              [clojure.java.browse :refer [browse-url]]))
004  
005  (defn browse-image [file-name]
006    (browse-url (str "file://" file-name)))
007  
008  (defn view-image [file-name]
009    (case (System/getProperty "os.name")
010      ("Mac OS X")
011      (sh "open" "-a" "Google Chrome" file-name)
012  
013      :otherwise
014      (browse-image file-name)))
