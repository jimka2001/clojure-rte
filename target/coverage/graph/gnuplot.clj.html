010  (defn run-gnu-plot [terminal gnu-name output-file-name]
011    (assert (string? output-file-name))
012    (assert (string? gnu-name))
013    (assert (string? terminal))
014    (sh "rm" "-f" output-file-name)
015    (let [s (sh "gnuplot" "-e" (str "set terminal " terminal ";"
016                                    "set output '" output-file-name "'"
017                                    )
018                gnu-name :in-enc "en_US.UTF-8" :out "en_US.UTF-8")]
019      (:exit s)))

022  ;;           ["plot-2" [ 2.1 3.3 4.5] [10.2 2.3 5.1]]
023  ;;           ["plot-3" [ 1.1 4.3 9.5 10.2] [6.2 2.3 7.1 7.5]]]
024  ;;          :view true
025  ;;          :title "sample plots"

028  (defn gnuplot [data-to-plot
029                 ;; either Seq[[String Seq[Number] Seq[Number]]]
030                 ;; or     Seq[[String Seq[[Number Number]]]
031                 & {:keys [terminals title comment x-axis-label x-log y-axis-label y-log grid
032                           output-file-base-name
033                           dir
034                           plot-with
035                           point-size
036                           key
037                           gnu-file-CB
038                           verbose
039                           view]
040                    :as gdp
041                    :or {terminals #{"png"}
042                         title ""
043                         comment ""
044                         x-axis-label ""
045                         x-log false
046                         y-axis-label ""
047                         y-log false
048                         grid false
049                         output-file-base-name "curves"
050                         dir "/tmp"
051                         plot-with "linespoints"
052                         point-size 0.8
053                         key "horizontal bmargin"
054                         gnu-file-CB (fn [_str])
055                         verbose false
056                         view false}}]
057    (let [curves (for [data data-to-plot] ;;  convert data-t-plot to the form Seq[(String,Seq[Double],Seq[Double])]
058                   (if (= 2 (count data))
059                     ;; [string [[x,y],[x,y],[x,y]...]]
060                     (let [[label seq-of-pairs] data]
061                       [label (map first seq-of-pairs) (map second seq-of-pairs)])
062                     data))
063          gnu-name (str dir "/" output-file-base-name ".gnu")
064          gnu (io/writer gnu-name)
065          write (fn [msg]
066                  (.write gnu msg)
067                  (.write gnu "\n"))
068          log-compatible (fn [projection]
069                           (every? (fn [data] 
070                                     (let [numbers (projection data)]
071                                       (and (every? #(> % 0) numbers)
072                                            (not-empty numbers)
073                                            (> (reduce max numbers)
074                                               (reduce min numbers)))))
075                                   curves))
076          ]
077      (assert (member plot-with ["linespoints" "points" "lines"]))
078      (assert gnu)
079      
080      (write (str "# " comment))
081      (if (and x-log
082               (log-compatible #(nth % 1)))
083        (write "set logscale x"))
084      (if (and y-log
085               (log-compatible #(nth % 2)))
086        (write "set logscale y"))
087      (if (not= "" x-axis-label)
088        (write (format "set xlabel '%s' font ',10'" x-axis-label)))
089      (if (not= "" y-axis-label)
090        (write (format "set ylabel '%s' font ',10'" y-axis-label)))
091      (if grid
092        (write "set grid"))
093      (write "set key font ',10'")
094      (write "set xtics font ',10'")
095      (write "set ytics font ',10'")
096      (if (= "points" plot-with)
097        (doseq [i (range (count curves))]
098          (write (format "set style line %d pt 7 ps %s" (inc i) point-size))))
099      
100      (write (format "set key %s" key))
101      (if (not= "" title)
102        (write (format "set title '%s' font ',12'" title)))
103      (.write gnu "plot ") ;; no \n
104      (let [footer (with-outstring pr-footer
105                     (let [header (with-outstring pr-header
106                                    (letfn [(write-curve [curve-title i xys]
107                                              (pr-header "'-' using 1:2")
108                                              (pr-header (format " with %s" plot-with))
109                                              (if (= "points" plot-with)
110                                                (pr-header (format " ls %d" i)))
111                                              (pr-header (format " title '%s'" curve-title))
112                                              (pr-footer (format "# %s\n" curve-title))
113                                              (doseq [[x y] xys]
114                                                (pr-footer (format "%.3f %.3f\n" (float x) (float y))))
115                                              (pr-footer "end\n"))]
116                                      (if (not-empty curves)
117                                        (do (let [[curve-title xs ys] (first curves)]
118                                              (write-curve curve-title 1 (map (fn [x y] [x y]) xs ys)))
119                                            (doseq [[i [curve-title xs ys]] (map-indexed vector (rest curves))]
120                                              (pr-header ",\\\n    ")
121                                              (write-curve curve-title (inc i) (map (fn [x y] [x y]) xs ys)))))))]
122                       (write header)))]
123        (write footer))
124      (.close gnu)
125      (gnu-file-CB gnu-name)
126      
127                       
128      (doseq [terminal terminals
129              :let [output-file-name (str output-file-base-name "." terminal)]]
130        (run-gnu-plot terminal gnu-name output-file-name)
131        (if view
132          (view-image output-file-name)))))

134  (defn histogram [buckets
135                   & {:keys [base-name
136                             dir
137                             x-label
138                             y-label
139                             title
140                             gnu-file-CB
141                             keep-if
142                             other-label
143                             view]
144                      :or {base-name "histogram"
145                           dir "/tmp"
146                           x-label "x"
147                           y-label "y"
148                           title ""
149                           gnu-file-CB (fn [_str])
150                           other-label "other"
151                           view false}}]
152    (assert (not (str/index-of x-label "\"")))
153    (assert (not (str/index-of y-label "\"")))
154    (assert (not (str/index-of x-label "\\")))
155    (assert (not (str/index-of y-label "\\")))
156    (let [png-file-name (str dir "/" base-name ".png")
157          gnu-header (cl-format false
158                                "~&~
159                                 set boxwidth 0.9 absolute~@
160                                 set style fill solid 1.00 border lt -1~@
161                                 set style histogram clustered gap 5 title textcolor lt -1~@
162                                 set style data histograms~@
163                                 set key outside top center horizontal~@
164                                 set xtics rotate by -45~@
165                                 set grid~@
166                                 set xlabel \"~A\"~@
167                                 set ylabel \"~A\"~@
168                                 set title \"~A\" font ',10'~@
169                                 "
170                                x-label
171                                y-label
172                                title)
173          line-1 "plot $MyData using 2:xtic(1) ti col,"
174          lines (for [[i _ign] (rest (map-indexed vector buckets))]
175                  (format "   $MyData using %d ti col," (+ 2 i)))
176          gnu-footer (str/join " \\\n" (cons line-1 lines))
177          gnu-file-name (str base-name ".gnu")
178          gnu (io/writer gnu-file-name)
179          write (fn [msg]
180                  (.write gnu msg))
181          data (for [[label counts] buckets
182                     :let [_ (println [:counts counts])
183                           fqs (sort-by first (into [] (frequencies counts)))
184                           _ (println [:fqs fqs])
185                           xys (for [[state-count num-samples] fqs
186                                     :when (keep-if state-count)
187                                     :let [percentage (/ (* 100.0 num-samples) (count counts))]]
188                                 [[state-count] percentage])
189                           _ (println [:xys xys])
190                           left-over-percent (- 100.0 (reduce + (map second xys)))]]
191                 [label (concat (sort-by first xys)
192                                [[[] left-over-percent]])])
193          counts (concat (map (fn [c] [c])
194                              (sort (distinct (for [[label xys] data
195                                                    [col percent] xys
196                                                    c col]
197                                                c))))
198                         [[]])
199          ]
200      (assert gnu (format "cannot open %s" gnu-file-name))
201  
202      (write gnu-header)
203      (write "\n\n")
204      (write "$MyData << EOD\n")
205      (write (format "\"%s\"" x-label))
206      (doseq [[label counts] buckets
207              :let [num-samples (count counts)]]
208        (write (format " \"%s samples=%d\"" label num-samples)))
209      (write "\n")
210                   
211      (doseq [sc1 counts
212              :let [label (if (empty? sc1)
213                            other-label
214                            (format "%s" (first sc1)))
215                    percentages (for [[_bucket_label xys] data
216                                      [sc2 percent] xys
217                                      :when (= sc2 sc1)]
218                                  (format "%.3f" percent))
219                    text (str/join " " percentages)]]
220        (write (format "\"%s\" %s\n" label text)))
221      (write "EOD\n\n")
222      (write gnu-footer)
223      (write "\n")
224      (.close gnu)
225      (gnu-file-CB gnu-file-name)
226      (run-gnu-plot "png" gnu-file-name png-file-name)
227  
228      (if view
229        (view-image png-file-name))))
