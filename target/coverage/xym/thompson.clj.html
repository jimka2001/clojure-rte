219  (defn remove-epsilon-transitions [in out transitions ;; List[Tuple[int, Optional[SimpleTypeD], int]]
220                                    ] ;; -> Tuple[int, List[int], List[Tuple[int, SimpleTypeD, int]]]
221    (let [[epsilon-transitions normal-transitions]
222          (partition-by-pred (fn [[_ tr _]] (= :epsilon tr)) transitions)
223          all-states (vec (find-all-states transitions))]
224      (letfn [(reachable-from [q]
225                (set (for [[x _ y] epsilon-transitions
226                      :when (= x q)]
227                  y)))
228              (extend-closure [m] ;; m is seq of set of int
229                (for [qs m]
230                  (set/union qs
231                             (reduce (fn [acc q]
232                                       (set/union acc (reachable-from q)))
233                                     #{}
234                                     qs))))]
235        (let [epsilon-closure (fixed-point (map (comp set list) all-states)
236                                           extend-closure 
237                                           =)
238              transitions2 (mapcat (fn [q closure]
239                                     (for [c closure
240                                           [x label y] normal-transitions
241                                           :when (= x c)
242                                           :when (not= x q)]
243                                       [q label y]))
244                                   all-states epsilon-closure)
245              ;; some states no longer exist after removing epsilon transitions
246              updated-transitions (concat normal-transitions transitions2)
247              remaining-states (find-all-states updated-transitions)
248              finals (mapcat (fn [q closure]
249                               (if (and (or (member q remaining-states)
250                                            (= q in))
251                                        (member out closure))
252                                 [q]))
253                             all-states epsilon-closure)]
254          (trim in finals updated-transitions)))))

263  (defn complete
264    "Transform a sequence of transitions (either deterministic or otherwise)
265     into a set of completed transitions.  I.e., the union of the types labeling
266     any arbitrary transition is STop.  In the case when it cannot be determined
267     whether the transitions are already locally complete, an addition transition
268     is added with is the complement of the existing transitions.  This may be
269     empty but if it is empty, .inhabited returns None (dont-know)."
270  
271    [in outs
272     clean-transitions ;; : List[Tuple[int, SimpleTypeD, int]]
273     ] ;; -> List[Tuple[int, SimpleTypeD, int]]:
274    (let [all-states (find-all-states clean-transitions)
275          sink (delay (new-state))
276          grouped (group-by first clean-transitions)
277          cf (fn [q]
278               (let [trs (get grouped q [])]
279                 (if (empty? trs)
280                   ;; for states with no exiting transitions
281                   [[q :sigma @sink]]
282                   ;; for states withs ome exiting transitions, find the complement of their union
283                   (let [tds (map second trs)
284                         remaining (template (not ~(gns/create-or tds)))]
285                     ;; if either satisfiable is Some(True) or dont-know None,
286                     ;; then we have to create a transition to the sink state
287                     (if (= false (gns/inhabited? remaining :dont-know))
288                       []
289                       [[q remaining @sink]])))))
290          completing-transitions (mapcat cf all-states)]
291      (cond (empty? clean-transitions)
292            [[in :sigma @sink]
293             [@sink :sigma @sink]]
294  
295            (empty? completing-transitions)
296            clean-transitions
297  
298            :else
299            (concat (conj completing-transitions
300                          [@sink :sigma @sink])
301                    clean-transitions))))

303  (defn determinize [in outs
304                     transitions ;; List[Tuple[int, SimpleTypeD, int]]
305                     ] ;; -> Tuple[int, List[int], List[Tuple[int, SimpleTypeD, int]]]:
306    ;; This can be done with a call to traceTransitionGraph.
307    ;;   to generate a graph (list of transitions) whose vertices are
308    ;;   each a Set[int].  Then you'll have to renumber back to
309    ;;   states as int.
310    ;;   BTW, you can also use traceTransitionGraph to implement
311    ;;   the cartesian product needed for And, as well as to remove
312    ;;   non-accessible, and non-coaccessible vertices.
313    ;; Use the mdtd function to partition a set/list of types into
314    ;;   a set of disjoint types.
315    (let [grouped (group-by first transitions)]
316      (letfn [(expand-one-state [qs] ;; qs is a set of int
317                ;; Given a set of states, find all the SimpleTypeD which are labels of
318                ;;   transitions leaving any of those states.
319                (assert (set? qs) (cl-format false "expecting a set, not ~A" qs))
320                (let [tds (set (for [q qs                                   
321                                     [_ td _] (get grouped q [])]
322                                 td))
323                      ;; The set tds of SimpleTypeD might contain some non-disjoint types.
324                      ;;   So we call compute the minimum disjoint type decomposition (mdtd)
325                      ;;   and create one transition per type per type from this decomposition.
326                      ;;   This is done in two steps.
327                      ;;   step 1. compute tr2 which is the list of (type,next-state) pairs.
328                      ;;   step 2. for { (td,pairs) ...} compute the seq of next states corresponding
329                      ;;      to each type.
330                      ;;      The pair returned from step 2 is exactly what is needed by traceGraph
331                      tr2 (for [{:keys [td factors]}  (gns/mdtd tds)
332                                q qs
333                                [_ td1 y] (get grouped q [])
334                                :when (member td1 factors)]
335                            [td y])]
336                  (for [[td pairs] (group-by first tr2)
337                        :let [next-states (set (map second pairs))]]
338                    [td next-states])))]
339  
340        (let [inX #{in}
341              [expanded-finals expanded-transitions]
342              (trace-transition-graph inX expand-one-state
343                                      (fn [qs]
344                                        (exists [f outs]
345                                                (member f qs))))]
346          ;; we now have transitions which map Set[Int] to Set[Int]
347          ;;   where each state (Set[Int]) represents 1 state.  We must
348          ;;   now replace each Set[Int] with a corresponding Int.
349          (renumber-transitions inX expanded-finals expanded-transitions)))))

353          completed (complete in2 outs2 clean)]
354      (determinize in2 outs2 completed)))
