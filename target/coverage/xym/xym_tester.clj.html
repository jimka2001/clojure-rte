012  (defn build-state-map
013    "This is a helper function for use in `gen-dfa` and is not intended
014    for general-purpose use.
015    `transitions` is a list of triples of the form [origin-index type-designator target-index]
016    We generate a map index->State as needed for the :states argument of map->Dfa.
017    However, the given transitions might not specify non-deterministic transitions.
018    Therefore we produce a new set of transitions that are deterministic.
019    E.g., if we have ([0 x 1]
020                      [0 y 2]
021                      [0 z 3])
022    Then we produce ([0 x 1]
023                     [0 (:and y (:not x)) 2]
024                     [0 (:and z (:not (:or x y)))])
025    Sometimes this :and/:not intersection creates a type designator which is know
026    to be vacuous, in which case the transition is omitted.
027    For this reason the number of transitions might be reduced,
028    and the number of transitions requested in the call to `gen-dfa`
029    will fail to be satisfied.
030    "
031    [transitions initial accepting]
032    (let [grouped (group-by first transitions)
033          state-ids (distinct (concat (map first transitions)
034                                      (map (fn [triple] (nth triple 2)) transitions)))]
035      (letfn [(mutually-exclusive [pairs] ;; pairs of the form [td target-id]
036                (loop [pairs pairs
037                       acc []]
038                  (let [td (delay (apply gns/And-not (cons (first (first pairs))
039                                                          (map first (rest pairs)))))]
040                    (cond (empty? pairs)
041                          acc
042  
043                          (gns/inhabited? @td :dont-know)
044                          (recur (rest pairs)
045                                 (conj acc
046                                       [@td (second (first pairs))]))
047  
048                          :else
049                          (recur (rest pairs) acc)))))]
050  
051        (into {} (for [origin-index state-ids
052                       :let [triples (get grouped origin-index [])
053                             mut-ex (mutually-exclusive (map rest triples))
054                             ]]
055                   [origin-index
056                    (xym/map->State {:index origin-index
057                                     :initial (= origin-index initial)
058                                     :accepting (member origin-index accepting)
059                                     :transitions mut-ex})])))))

065  (defn gen-dfa
066    "Generate a random Dfa having a specified number of states `num-states` and approximately
067    `num-transitions` number of transitions.
068    `num-transitions` requested number of total transitions.  This might fail to be
069    satisfied in the case that unsatisfiable transitions are produced during
070    the determinization loop (see `build-state-map`).
071    "
072    [& {:keys [num-states num-transitions exit-value type-size probability-indeterminate]
073        :or {num-states 10
074             num-transitions 20
075             exit-value true
076             type-size 2
077             probability-indeterminate 0.15}}]
078    (letfn [(choose-type []
079              (if (< (rand) probability-indeterminate)
080                (gen-indeterminate-type (* 2 type-size))
081                (gen-inhabited-type type-size)
082                ))]
083      (let [tr1 (- num-transitions num-states)
084            transitions (concat (for [k (range tr1)
085                                      :let [origin (rand-int (dec num-states))
086                                            ;; transition satisfiability is either true or :dont-know
087                                            ;;  not false.
088                                            td (choose-type)
089                                            target (rand-int (dec num-states))
090                                            ]]
091                                  [origin td target])
092                                ;; generate path from 0 to (num-states - 1)
093                                (for [id (range (dec num-states))]
094                                  [id
095                                   (choose-type)
096                                   (inc id)]))]
097        (xym/map->Dfa 
098         {:exit-map {:default exit-value}
099          :combine-labels gns/combine-labels
100          ;; create map index -> State
101          ;; divide transitions into two parts one transition for each state to some later state
102          ;; after that transitions between randomly selected states
103          :states (build-state-map transitions
104                                   0
105                                   [(dec num-states)])}))))
