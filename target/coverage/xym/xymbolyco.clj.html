040  (defrecord State 
041      ;; :index -- the index of this state in the array
042      ;; :accepting - Boolean true/false indicating whether this state is a
043      ;;     final/accepting state.
044      ;; :initial - Boolean indicating initial state
045      ;; :pattern -- the derivative value representing an rte pattern matching
046      ;;     any tail of the input sequence which is accepting from this point
047      ;;     onward.
048      ;; :transitions -- A list of pairs, each pair is a 2 element array of the form
049      ;;     [type next-state], e.g., [clojure.lang.Keyword 1]
050      ;;     which means if the value at the head of the sequence is of type
051      ;;     clojure.lang.Keyword, then go to state 1.  The type is some type designator
052      ;;     as specified in the genus library.  the state index is some index of the state
053      ;;     array representing the finite atomaton.
054      [index description initial accepting pattern transitions])

065  (defn exit-value
066    "Given a Dfa and either a State or state-id (integer), compute the exit value of
067    the state by calling the function :exit-map in the dfa.
068    The exit map might be a map whose keys are the state indices (integers)
069    in which case we look up the associated value.
070    However, the index might be missing, in which case we look for the
071    key :default, else we return nil"
072    [dfa state]

074      (exit-value dfa (:index state))

076       ;; if there is no such index in the exit-map, then we look for
077       ;;   the key :default
078       ((:exit-map dfa) :default nil))))

112  (defn check-dfa
113    "assert that no transition references an invalid state"
114    [dfa]
115    (assert (instance? Dfa dfa))
116    (assert (:combine-labels dfa) (format "missing :combine-labels in Dfa %s" dfa))
117    (assert (map? (:states dfa))  (cl-format false "states must be a map, not a ~A: ~A" (type (:states dfa)) (:states dfa)))
118    (let [ids (set (ids-as-seq dfa))]
119      (doseq [id (keys (states-as-map dfa))
120              q (get (states-as-seq dfa) id)]
121        (assert (instance? State q) (cl-format false "expecting a State, got a ~A, ~A" (type q) q))
122        (assert (:index q) (format "state %s has emtpy :index" q))
123        (assert (= q (state-by-index dfa (:index q)))
124                (format "state %s disagrees with its index %s" q (:index q)))
125  
126        (assert (= (:transitions q)
127                   (distinct (:transitions q)))
128                (cl-format false "~A transitions contains a duplicate: ~A"
129                           q (:transitions q)))
130        (doseq [[label-1 dst-1] (:transitions q)
131                [label-2 dst-2] (:transitions q)
132                :when (not (= [label-1 dst-1]
133                              [label-2 dst-2]))]
134          (assert (bdd/type-disjoint? label-1 label-2)
135                  (cl-format false "overlapping types ~A vs ~A in ~A transitions ~A"
136                             label-1 label-2 q (:transitions q))))
137        (doseq [:let [trans-labels (map first (:transitions q))]
138                [label freq] (frequencies trans-labels)]
139          (assert (= 1 freq)
140                  (cl-format false "label ~A appears ~D times in transitions of ~A: transitions=~A"
141                             label freq q (:transitions q)))))
142  
143      (doseq [q (states-as-seq dfa)
144              [label dst-id] (:transitions q)]
145        (assert (member dst-id ids) (format "transition %s leads to invalid state: states are %s"
146                                            [label dst-id]
147                                            ids)))
148      dfa))

174    "This is a helper function used by optimized-transition-function.
175    gen-funciton computes and returns a unary function which can be called
176    on a candidate element coming from an input sequence, and returns
177    the id of the next state."
178    [transitions promise-disjoint sink-state-id]
179    (letfn [(type-intersect [t1 t2]
180              (list 'and t1 t2))
181            (type-and-not [t1 t2]
182              (if (= :empty-set t2)
183                t1
184                (list 'and t1 (list 'not t2))))]
185      (let [state-id->pseudo-type (into {} (for [[_type state-id] transitions
186                                                 :let [tag (gensym "pseudo-")]]
187                                             [state-id (template (satisfies ~tag))]))
188            pseudos (for [[_ tag] state-id->pseudo-type]
189                      tag)
190            pseudo-type->state-id (into {} (for [[state-id tag] state-id->pseudo-type]
191                                             [tag state-id]))
192            pseudo-type-functions (for [[_ [_ function-designator]] state-id->pseudo-type]
193                                    function-designator)
194            old-label-< bdd/*label-<*
195            bdd (binding [gns/*pseudo-type-functions* (concat pseudo-type-functions
196                                                              gns/*pseudo-type-functions*)
197                          bdd/*label-<* (fn [t1 t2]
198                                          (cond (and (member t1 pseudos)
199                                                     (member t2 pseudos))
200                                                (old-label-< t1 t2)
201  
202                                                (member t1 pseudos)
203                                                false
204  
205                                                (member t2 pseudos)
206                                                true
207  
208                                                :else
209                                                (old-label-< t1 t2)))]
210                  (first
211                   (reduce
212                    (fn [[accum-bdd previous-types] [type state-id]]
213                      [(bdd/or accum-bdd
214                               (bdd/bdd (type-and-not
215                                         (type-intersect type
216                                                         (state-id->pseudo-type state-id))
217                                         (if promise-disjoint
218                                           ;; This is an optimization
219                                           ;; see issue
220                                           ;; https://gitlab.lrde.epita.fr/jnewton/clojure-rte/-/issues/27
221                                           ;; If the given types are already promised to be disjoint,
222                                           ;;  then no need to do an expensive Bdd operation
223                                           :empty-set
224                                           ;; in case the types are not disjoint
225                                           (cons 'or previous-types)))))
226                       (cons type previous-types)])
227                    [false '(:empty-set)] ;; initial bdd and empty-type
228                    transitions)))]
229                  ;;(clojure-rte.dot/bdd-to-dot bdd :title (gensym "bdd") :view true)
230        (fn [candidate]
231          (loop [bdd' bdd
232                 lineage ()]
233            (cl/cl-cond
234             ((member bdd' '(true false))
235                        ;; transitions not exhaustive
236              sink-state-id)
237             ((pseudo-type->state-id (:label bdd') false)) ;; this is the return value of the (fn [] ...)
238             ((gns/typep candidate (:label bdd'))
239              (recur (:positive bdd')
240                     (cons (:label bdd') lineage)))
241             (true
242              (recur (:negative bdd')
243                     (cons (list 'not (:label bdd')) lineage)))))))))

245  (def optimized-transition-function
246    "Given a set of transitions each of the form [type-designator state-index],
247    return a unary transfer function which can be called with an candidate element
248    of a sequence, and the function will return the id of the next state.  When called
249    with the candidate object, will not evaluate any type predicate more than
250    once. optimized-transition-function also accepts a sink-state-id which is
251    what the transfer function will return if no transition label matches the
252    candidate element."
253    
254    ;; TODO, if we could know whether to trust that the transitions are
255    ;;  already disjoint this function could be made much faster.
256    ;;  It is not necessary to know whether the transitions cover the
257    ;;  universe because the indicator function has a second argument
258    ;;  to return if there is no match.
259    (gc-friendly-memoize
260      (fn [transitions promise-disjoint sink-state-id]
261    (bdd/with-hash []
262      (letfn [
263              ;; local function find-duplicates
264              (find-duplicates [items]
265                (loop [items items
266                       duplicates []]
267                  (cond (empty? items)
268                        (distinct duplicates)
269  
270                        (member (first items) (rest items))
271                        (recur (rest items)
272                               (conj duplicates (first items)))
273  
274                        :else
275                        (recur (rest items)
276                               duplicates))))
277  
278              ]
279        (let [types (map first transitions)
280              duplicate-types (find-duplicates types)
281              inhabited-types (delay (filter (fn [td] (gns/inhabited? td false))
282                                             types))
283              consequents (map second transitions)]
284          
285          (cond
286            (not= (count consequents) (count (distinct consequents)))
287            ;; If there is a duplicate consequent, then the corresponding types can be unioned.
288            (optimized-transition-function (for [[consequent transitions] (group-by second transitions)]
289                                              [(gns/create-or (map first transitions)) consequent])
290                                            promise-disjoint
291                                            sink-state-id)
292  
293            (empty? duplicate-types)
294            (gen-function transitions promise-disjoint sink-state-id)
295            
296            (not-empty (intersection duplicate-types inhabited-types))
297            ;; if some duplicate types are inhbited
298            (throw (ex-info (cl-format false "transitions ~A has a duplication of types: ~A"
299                                       transitions (find-duplicates types))
300                            {:transitions transitions
301                             :duplicates (find-duplicates types)}))
302  
303            ;; if all duplicate types are empty types
304            :else
305                             (gen-function transitions promise-disjoint sink-state-id))))))))

345  (defn find-hopcroft-partition
346    "Apply the Hopcroft partition algorithm to the states of the given
347    Dfa to return a set of eqv-classes.  This set of eqv-classes is a partition
348    of the initial set of states.
349    Each eqv-class is a set of states.
350    This function returns a sequence of sets, where each set contains
351    State's, not state-id's, E.g., ( ... #{#<State 9> #<State 2>} #{#<State 0} ...),
352    not (... #{9 2} #{0} ...)"
353    [dfa]
354    (let [[finals non-finals] (map (group-by (comp boolean :accepting)
355                                             (states-as-seq dfa)) [true false])
356          pi-0 (conj (split-eqv-class finals
357                                      (fn [state]
358                                        (exit-value dfa state)))
359                     non-finals)]
360      (letfn [(refine [partition]
361                (letfn [(phi [source-state label]
362                          ;; find the element of partition, itself an equivalence class, which
363                          ;;   which contains the destination state of the transition
364                          ;;   whose source and label are given
365                          (find-eqv-class partition (delta dfa source-state label)))
366                        (Phi' [s]
367                          (for [[label _dst-id] (:transitions s)]
368                            [label (phi s label)]))
369                        (Phi [s]
370                          (for [[k pairs] (group-by second (Phi' s))
371                                :let [labels (map first pairs)
372                                      label (reduce (:combine-labels dfa) labels)]]
373                            [label k]))
374                        (repartition [eqv-class]
375                          (split-eqv-class eqv-class Phi))]
376                  (mapcat repartition partition)))]
377        (fixed-point pi-0 refine =))))

876  (defn find-spanning-map
877    "Return a map from exit-value to [satisfiability path],
878    where `satisfiability` is one of ...
879    and `path` is is a list from initial state to final state"
880    [dfa]
881    (let [states (states-as-seq dfa)
882          [f indeterminate-weight] (gen-dijkstra-edges dfa)
883          q0 (initial-state dfa)
884          [d p] (dijkstra (:index q0) f)
885          ]
886      ;; dijkstra has computed the shortest paths from state 0
887      ;;   to every other state in the dfa.
888      ;;   1. we are only interested in accepting states
889      ;;   2. if two accepting states have the same exit-value
890      ;;       then we choose the minimum of the two paths.
891      (reduce (fn [acc [ev states]]
892                ;; `states` is a seq of States, we want to choose the one
893                ;;   with the minimum distance as per the `d` map.
894                (let [best-q (apply min-key (fn [q] (get d (:index q))) states)
895                      best-path-delayed (get p (:index best-q))
896                      best-path (when best-path-delayed
897                                  @best-path-delayed)
898                      satisfiability (if (< (get d (:index best-q))
899                                            indeterminate-weight)
900                                       :satisfiable
901                                       :indeterminate)]
902                  (if best-path
903                    (assoc acc ev [satisfiability best-path])
904                    acc)))
905              {}
906              (group-by (fn [q] (exit-value dfa q))
907                        (filter :accepting states)))))

910    "Return a map from exit-value to [satisfiability path]
911    where path is a sequence of pairs [satisfiability type-designator]"
912    [dfa]

914              (case (gns/inhabited? td :dont-know)

927                                 [(sat-label td)

930                  )))
931            ]

938    "Return pair of the form [key path]
939    where key is one of :unsatisfiable, :satisfiable, :indeterminate
940    and path is a list of of integers, each an index of a state,
941    the path leads from initial state to final state."
