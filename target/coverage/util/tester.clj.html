028  (defn simplify [unary error-case gen-components]
029    (try (unary error-case)
030         error-case
031         
032         (catch Exception e
033           (cl-format true "e=~A~%" e)
034           (or (some (fn [component]
035                       (simplify unary component gen-components)) (gen-components error-case))
036               error-case))))
037  
038  (defn de-lazify [obj]
039    (cond (not (sequential? obj))
040          obj
041  
042          (empty? obj)
043          obj
044  
045          :else
046          (map-eagerly de-lazify obj)))

048  (defn random-test
049    "Call a testing function, unary-test-fun, on *randomly* generated values.
050    unary-test-fun is a unary function which can be called with the return value of
051       arg-generator as sole argument.   If the function returns, it is deemed successful.
052       If the function needs to indicate failure, it should throw an exception as if by
053       (assert false \"some error message\").
054    arg-generator is called num-tries number of times to generate input for
055    num-tries number of calls to function unary-test-fun.
056    gen-components is currently unused, but in the future will be used to simplify expression
057        to attempt to generate a simpler test case, if test fails.
058    verbose indicates whether to print verbose information about progression of tests."
059    [num-tries unary-test-fun arg-generator _gen-components verbose]
060    (doseq [n (doall (range num-tries))
061            :let [data (de-lazify (arg-generator))]]
062      (when verbose
063        (cl-format true "~d/~d: trying ~A~%" n num-tries data))
064      
065      (try (unary-test-fun data)
066           (catch java.lang.StackOverflowError e
067             (cl-format true "~&Stack overflow on ~A~%" data)
068             (throw e)))))
