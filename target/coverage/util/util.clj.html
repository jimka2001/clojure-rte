039  (defn remove-once 
040    "Non-destructively remove the first element of the sequence which is
041     = to the target value.  The list is unrolled as much as necessary,
042     to remove the target value, and then the leading values are
043     prepended, via concat, to the beginning of the remaining sequence.
044     The tail of the sequence after finding the target is not examined,
045     in case it is lazy.  If the taget does not appear in the list, a
046     copy of the sequence is returned.  If the target item appears more
047     than once, we have no way of knowing, and only the first such
048     occurance is removed."
049    [target items]
050  
051    (loop [items items
052           acc ()]
053      (cond
054        (empty? items)
055        (reverse acc)
056  
057        (= (first items) target)
058        (concat (reverse acc) (rest items))
059  
060        :else
061        (recur (rest items) (cons (first items) acc)))))

216  (defn find-simplifier
217    "Iterate through a sequence of so-called simplifiers.  Each simplifier
218    is a unary function which accepts the given obj as arguments.  Each
219    simplifier is expected to either _simplify_ the object or leave it as-is.
220    If the simplifier returns an object which is = obj, then we assume the
221    simplifier chose NOT to simplify, in which case find-simplifier proceeds
222    to the next simplifier in the sequence.
223    If, however, the simplifier returns an object which is not = to obj, 
224    then we assume that the simplifier HAS simplified the object.
225    WARNING, find-simplifier does not in any way verify that the object
226    is more simple than before; it only checks whether it is different.
227    Once a simplifier has _simplified_ an object, find-simplifier returns
228    the newly generated object, and the remaining simplifiers are silently
229    ignored."
230    ([obj simplifiers]
231     (find-simplifier obj simplifiers false))
232    ([obj simplifiers verbose]
233     ;;(cl-format true "~A ~A~%" (stacksize) obj)
234     (if (empty? simplifiers)
235       obj
236       (loop [[f & fs] simplifiers
237              i 0]
238         (let [new-obj (f obj)]
239           (cond
240             (not= new-obj obj)
241             (do
242               (when verbose
243                 (cl-format true "[~A] ~A~%   --> ~A~%" i obj new-obj))
244               new-obj)
245  
246             (empty? fs)
247             obj
248  
249             :else
250             (recur fs (inc i))))))))

254    "Find the fixed point of the given function starting at the given value.
255    Convergence is detected when the function good-enough returns Boolean true
256    when called with two consecutive values. The older of the two values
257    is returned in this case."
258    ([value f good-enough]
259     (fixed-point value f good-enough false))
260    ([value f good-enough verbose]
261    (loop [value value
262           old-values ()]
263      (let [new-value (f value)]
264        (when verbose
265          (cl-format true "reducing ~A~%" value)
266          (cl-format true "   to -> ~A~%" new-value))
267        (cond (good-enough value new-value)
268              value
269  
270              (member value old-values)
271              (throw (ex-info (cl-format nil "infinite loop in fixed-point, twice encountered ~A" value)
272                              {:value value
273                               :old-values old-values}))
274  
275              :else
276              (recur new-value (cons value old-values)))))))

404  (defn lazy-pairs
405    ;; code provided by Eugene Pakhomov
406    ;; https://app.slack.com/client/T03RZGPFR/C03S1KBA2/user_profile/U2FRKM4TW
407    "Generate (lazily) a sequence of pairs (a b) from the given sequence such
408    that a is always to the left of b in the given sequence.   Supposing that
409    the given sequence has no duplicates (1 2 3) -> ((1 2) (1 3) (2 3)).
410    The order of the list of pairs is not guaranteed, however the following
411    property holds.
412    At the point in the sequence where [a b] occurs, every pair [x y] thereafter
413    has the property that either max(x,y) >= max(a,b). E.g every pair containing
414    both entries less than 13 appears before the first occurance of 13.
415    If the sequence contains [4 13], then every element [a b] to the left has
416    both a and b <= 13 and no [a b] after [4 13] has both a and b < 13.
417    This ordering property ensures that if the given collection is a lazy
418    sequence, that as few items as possible are realized.  I.e., if the first
419    n items have been realized, then all pairs involving the first n items
420    have been generated before the n+1'th item gets realized."
421    ([coll]
422     (lazy-pairs (list (first coll)) (next coll)))
423    ([seen coll]
424     (when (seq coll)
425       (lazy-seq
426         (let [x (first coll)]
427           (concat (map (fn [s]
428                          [s x])
429                        seen)
430                   (lazy-pairs (cons x seen) (next coll))))))))

666  (defn trace-graph [v0 edges]
667    ;; The loop traverses the graph generated by the function, edges.
668    ;; The traversal is in numerical order.  The initial vertex is 0, the next
669    ;; vertex generated is 1.  Once the edges of vertex 0 are generated, several
670    ;; vertices may have been generated, but vertex 1 is traversed next, then
671    ;; 2, 3, ...
672    ;; The variable, next-available-state, keeps track of the next available
673    ;;    vertex number.  When the edges function returns a new v which
674    ;;    has not yet been numbered, then it gets assigned next-available-state.
675    ;; next-available-state gets incremented (via recur) whenever a new vertex
676    ;;    is encountered, and associated with the current value of next-available-state.
677    ;; The recursion terminates when the index of v is == to next-available-state
678    ;;    because that means no new vertices have been encountered, thus
679    ;;    all vertices have been visited.
680    (loop [current-state-id 0 ; int
681           next-available-state 1 ; int
682           es (edges v0) ; seq of [L V] label vertex
683           int-to-v [v0] ; vector of V, i.e. map int to V
684           v-to-int {v0 0} ; map V to int
685           m [[]] ; vector of seq of [L int], i.e. map of int to seq of [L int]
686           ]
687      (if (empty? es)
688        ;; The termination condition occurs if next == next-available-state, because
689        ;;   this means now new vertices have been encountered which have not
690        ;;   yet been treated.
691        (let [next (inc current-state-id)]
692          (if (< next next-available-state)
693            (let [v2 (int-to-v next)]
694              (recur next next-available-state (edges v2) int-to-v v-to-int (conj m [])))
695            [int-to-v m]))
696        
697        (let [[[label v1] & lvs] es]
698          (if (not (get v-to-int v1 false))
699            ;; if we are seeing v1 for the first time, then register it
700            ;;   in int-to-v and in v-to-int.
701            (recur current-state-id (inc next-available-state) es
702                   (conj int-to-v v1) (assoc v-to-int v1 next-available-state) m)
703            (recur current-state-id next-available-state lvs int-to-v v-to-int
704                   (assoc m
705                          current-state-id
706                          (conj (m current-state-id) [label (v-to-int v1)]))))))))
