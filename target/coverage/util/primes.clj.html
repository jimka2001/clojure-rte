007  (defn prime-factorization
008    ([n]
009     (prime-factorization n 2 (sqrt n) []))
010    ([n lower upper factors]
011     (cond (> lower upper)
012           (conj factors n)
013  
014           (= 0 (mod n lower))
015           (let [n (/ n lower)]
016             (recur n lower (sqrt n) (conj factors lower)))
017  
018           :else
019           (recur n (inc lower) upper factors))))

023  (defn primes
024    ([n]
025     (assert (> n 1))
026     (primes n 3 [2]))
027    ([n lower found]
028     (cond (> lower n)
029           (into () (reverse found))
030  
031           (some (fn [p] (= 0 (mod lower p)))
032                 found)
033           (recur n (+ 2 lower) found)
034  
035           :else
036           (recur n (+ 2 lower) (conj found lower)))))

039  (defn encode-as-factorization [n]
040    (let [factorization (frequencies (prime-factorization n))
041          max-prime (reduce max (keys factorization))
042          primes (primes max-prime)]
043      (map #(get factorization % 0) primes)))

061    (cond (= a b)
062          true
063  
064          (and (not-empty a)
065               (not-empty b)
066               (< (first a) (first b)))
067          (recur (rest a) (rest b))

105  (defn topological-sort 
106    ([items a<b]
107     (topological-sort items a<b []))
108    ([items a<b sorted-so-far]
109  
110     (let [unconstrained (filter (fn [i] (not-any? (fn [[a b]] (= i b)) a<b)) items)
111           hit-count (bounded-count 2 unconstrained)]
112       (cond (empty? items)
113             [sorted-so-far :sorted]
114  
115             (= 0 hit-count)
116             [sorted-so-far :over-constrained] ;; no order exists
117  
118             (< 1 hit-count)
119             [sorted-so-far :under-constrained] ;; multiple orders exist
120  
121             :otherwise
122             (let [found (first unconstrained)
123                   items (filter #(not= % found) items)
124                   a<b (filter (fn [[a b]] (not= a found)) a<b)]
125  
126               (recur items a<b (conj sorted-so-far found)))))))
