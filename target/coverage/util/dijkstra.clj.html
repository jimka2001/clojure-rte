013  (defn remove-keys [m pred]
014    (select-keys m (remove pred (keys m))))
015  
016  (defn find-path 
017    "return a list starting with `start` ending with `v` which represents
018    the shortest path from `start` to `v` as indicated in the `p` map.
019    The map `p` is the precedence map created by the `dijkstra` function
020    and likewise by `dijkstra-to-final`."
021    [start v p]
022    (loop [v v
023           path (list)]
024      (if (= v start)
025        (conj path start)
026        (recur (get p v) (conj path v)))))

029  (defn dijkstra
030    "Computes single-source shortest path distances in a directed graph.
031  
032    Given a node n, (f n) should return a map with the successors of n
033    as keys and their (non-negative) distance from n as vals.
034  
035    Returns a vector with a map from nodes to their distance from start
036    and a map of vector to delay object, which will yield a path from
037    start to the node in question."
038    [start f]
039    (letfn [(f-no-self-loop [v] (dissoc (f v) v))]
040      ;; f-no-self-loop is now a function which wraps the given function, `f`
041      ;;   but removes self loops from its return values
042      (loop [q (priority-map start 0)
043             r {}
044             p {}]
045        (if-let [[v d] (peek q)]
046          (let [dist (-> (f-no-self-loop v) (remove-keys r) (update-vals #(+ d %)))
047                [q p] (reduce-kv (fn [[q p] n d]
048                                   (let [curr-d (q n)]
049                                     (if (or (nil? curr-d) (< d curr-d))
050                                       [(assoc q n d) (assoc p n v)]
051                                       [q p])))
052                                 [(pop q) p]
053                                 dist)]
054            (recur q (assoc r v d) p))
055          [r (into {} (for [k (keys p)]
056                            [k (delay (find-path start k p))]))]))))

059    "Like `dijkstra` function above, except that the search aborts
060    when a path is found to one of the given final states.
061    Returns [f d path] if the shortest path to a final state is to f, and
062    where d is the shortest distance.
063    path is ??? TODO document path"
064    [start f finals]
065    (letfn [(f-no-self-loop [v] (dissoc (f v) v))]
066      (loop [q (priority-map start 0)
067             r {}
068             p {}]

071            (not found)
