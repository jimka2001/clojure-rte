033  (defmacro cl-cond
034    "Like CL:cond.  Each operand of the cl-cond is a list of length at least 1.
035     The same semantics as clojure cond, in that the return value is
036     determined by the first test which returns non-false.  The
037     important semantic difference is that an agument has 1, then the
038     specified form is both the test and the return value, and it is
039     evaluated at most once.
040     Implementation from:
041     https://stackoverflow.com/questions/4128993/consolidated-cond-arguments-in-clojure-cl-style"
042    ([] nil)
043    ([[if1 & then1] & others]
044     (when (or if1 then1 others)
045       (let [extra-clauses# (when others `(cl-cond ~@others))]
046         (if then1
047           (case if1
048             (:else)
049             `(do ~@then1) ;; avoid silly lint error, lein eastwood
050  
051             (false nil)
052             `(do ~extra-clauses#)
053             
054             ;; else
055             `(if ~if1 (do ~@then1) ~extra-clauses#))
056           `(or ~if1 ~extra-clauses#))))))

058  (defn call-with-escape
059    "A functional interface to CL block/return.
060    The caller of call-with-escape provides a unary function.
061    The body of that function may call the function passed as argument.
062    to cause call-with-escape to return. E.g.,
063    (call-with-escape
064      (fn [ret1]
065        ;; now ret1 is a unary function, callint ret1 with argument x
066        ;; causes call-with-escape to immediately return x
067        (if something
068            (ret1 42) ;; return 42 from call-with-escape
069           ...)))"  
070    [unary]
071    (letfn [(ret [v]
072              (throw (ex-info "" {:data v
073                                  :ident ret})))]
074      (try (unary ret)
075           (catch clojure.lang.ExceptionInfo e
076             (if (= ret (:ident (ex-data e)))
077               (:data (ex-data e))
078               (throw e))))))

080  (defmacro with-escape [ret & body]
081    `(call-with-escape (fn [~ret] ~@body)))

083  (defn ldiff
084    "Returns a copy of the first prefix of l-left whose
085     tail is = (but perhaps not pointer-identical) to l-mid."
086    [l-left l-mid]
087    ;; we can't check pointer identity here because
088    ;;  the sequence might be lazy.
089    (loop [acc ()
090           l-iter l-left]
091      (cond (= l-mid l-iter)
092            (reverse acc)
093  
094            (empty? l-iter)
095            l-left
096  
097            :else
098            (recur (cons (first l-iter) acc)
099                   (rest l-iter)))))
