060  (defn clauses-to-dfa
061    "Returns a complete dfa which is the union of the input clauses.
062    E.g.,
063    (clauses-to-dfa [[0 rte-0] [1 rte-1] [3 rte-3] [2 rte-2] ...])
064    returns a dfa for which (rte/match dfa some-seq) will return 0 if
065    rte-0 is matched, 1 if rte-1 is matched etc...
066    If none of the rtes match, then false is returned.
067    "
068    [pairs]
069    (reduce xym/synchronized-union
070            (map (fn reduce-synchronized-union [[index rte]]
071                   (rte-to-dfa rte index))
072                 pairs)))

094  (defn rte-case-fn
095    "`pairs` is a set of pairs, each of the form [rte 0-ary-function]
096    This function is used in the macro expansion of rte-case but
097    can also be used on its own.
098    `rte-case-fn` returns a unary function which can be falled with
099    a sequence.  The sequence is matched against the rtes in order
100    and the first one that matches, the corresponding 0-ary-function
101    is called and its value is returned.
102    The sequence is traversed a maximum of once, but may not traverse
103    entirely if it is determined early that no rte matches."
104    [pairs code-exprs call-site-meta-data]
105    {:pre [(map? call-site-meta-data)]}
106    (let [dfa (reduce xym/synchronized-union 
107                      (for [[rte thunk] pairs]
108                        (do (assert (rte/valid-rte? rte)
109                                    (format "%s is not a valid rte" rte))
110                            (assert (fn? thunk))
111                            (rte-to-dfa rte thunk))))
112          traces (xym/find-spanning-map dfa)
113          ]
114      (when *warn-on-unreachable-code*
115        (doall (map (fn [[rte thunk] code-expr]
116                      (when (not (traces thunk))
117                        (print-unreachable-warning code-expr call-site-meta-data)))
118                    pairs code-exprs)))
119      ;; (dot/dfa-to-dot dfa :title (gensym "rte-case") :view true :draw-sink false)
120      (fn f-101 [s]
121        (rte/match dfa s))))

123  (defmacro rte-case
124    "Takes an expression, and a set of clauses.
125    The expression should evaluate to an object which is `sequential?`.
126    Each clause is of the form
127       rte consequent.
128    The rte is NOT evaluated, and should not be quoted.  It should be
129    a syntacticaly correct regular type expression.
130    If the first rte matches the sequence, then the consequent is evaluated
131    and its value is returned.   If the rte fails to match the rte, then
132    the second clause is examined.   The semantics are that the rte's are
133    considered in order, top to bottom, and the first one which matches
134    causes the corresponding consequent to be evaluated.
135    However, the sequence is traverse only once, so the matching process
136    is more efficient than a sequence of consecutive calls to
137    rte/match."
138    [sequence & clauses]
139  
140    ;; if odd number of clauses, then don't complain if the final
141    ;;   clause is unreachable
142    (let [[keywords clauses] (parse-prefixed-keyword-args clauses)
143          odd-clauses (odd? (count clauses))
144          clauses (if odd-clauses
145                    `(~@(butlast clauses)
146                      ~sigma-* ~(last clauses))
147                    clauses)
148          ;; This macro expansion may seem a bit more complicated that it
149          ;; needs to be.  I think there's a way to avoid the complication
150          ;; but I have thus far not been able to do so.
151          ;; We need the Dfa to be constructed only once.  This one-time construction
152          ;; is accomplished by the (let ...) in the macro expansion making
153          ;; a call to rte-case-fn, which is a memoized function.
154          ;; In order for the memoization to work the arguments passed to
155          ;; must be all static, i.e., containing no function objects.
156          ;; Therefore we cannot put (fn [...] ....) into the call rte-case-fn.
157          ;; Instead, we replace the clojures with indexes into an array
158          ;; of closures which also sits in the (let [...] ...)
159          ;; rte-case-fn returns a function which can be called with a sequence,
160          ;; and will return an index into the array.  The index is used to
161          ;; grab a 0-ary function from the array and call it.
162          parts (partition 2 2 clauses)
163          n-parts (count parts)
164          n-parts-to-check (if odd-clauses
165                             (dec n-parts)
166                             n-parts)
167          pairs (into [] (for [k (range n-parts)
168                               :let [[rte expr] (nth parts k)
169                                     name (symbol (format "rte-case-fn-%s" k))]]
170                           ;; we have to use ~'rte-fn because we don't
171                           ;; want backquote to slap a namespace on
172                           ;; the symbol
173                           `['~rte (fn ~name [] ~expr)]))
174          code-exprs (for [[_ expr] parts]
175                       expr)
176  
177          call-site-meta-data (merge (assoc (meta &form)
178                                            :file *file*)
179                                     keywords)
180          ]
181      
182      `(let [f# (rte-case-fn ~pairs '~code-exprs '~call-site-meta-data)
183             seq# ~sequence
184             thunk# (f# seq#)
185             ]
186         (if (fn? thunk#)
187           ;; f# has returned false if the sequence didn't match
188           ;; any of the rtes.  Otherwise it returned a 0-ary function,
189           ;; so we can call this 0-ary function.
190           (thunk#)
191           (throw (ex-info "No pattern matching given sequence"
192                           {:sequence seq#}))))))

194  (defn remove-extra-syntax 
195    "lambda-list is a vector which is almost compatible with the
196     lambda-list of fn.   However there might be a :allow-other-keys
197     in the map immediately of the &
198     E.g.,  [1 2 3 & {:keys [a b c] :allow-other-keys true}]
199     If such an element is found, return a copy of the vector
200        except with :allow-other-keys removed from the map
201      otherwise just return the vector as is."
202    [lambda-list]
203    (if (not (member '& lambda-list))
204      lambda-list
205      (let [[before after] (split-with (fn [x] (not= x '&)) lambda-list)]
206        (cond
207          (<= (count after) 1)
208          lambda-list
209          (not (map? (nth after 1)))
210          lambda-list
211          :else
212          (into [] (concat before
213                           ['&]
214                           [(dissoc (nth after 1) :allow-other-keys)]
215                           (drop 2 after)))))))

217  (defn lambda-list-to-rte
218    "Returns an rte either of one of the following forms:
219      (:cat ... (:* ...)) -- if the given lambda-list contains &
220      (:cat ...) -- if the given lambda list only has required parametes.
221    The lambda-list itself might contain type hints such as [^Boolean a b ^Long c],
222    and the types-map might contain type hints for the same or perhaps different variables
223    such as {b String c (satisfies odd?)}
224    If there are multiple type hints for the same variable, they both apply, e.g.,
225    in this example the type designator for c would be (and Long (satisfies odd?)).
226    If a type hint is given for an [& other] variable, then the type hint
227    implicitly means sequence of that type.  
228    e.g., [& other] {other Boolean}, means that other has type sequence of Boolean.
229    This corresponds to the equivalent of an rte such as (:* Boolean)
230  
231    The type-map may have keys corresponding to variables in the lambda list,
232    or keys of vectors of variables such as {a Boolean [b c] Number}.
233    In such a case the type designator corresponds equally to both variables.
234    In the case that a variable is mentioned stand-alone and also in a vector,
235    both type designators apply (intersection type).
236    "
237    [lambda-list types-map]
238    (assert (map? types-map))
239    (letfn [(expand-multi-restrictions [types-map]
240              (assert (map? types-map))
241              (merge types-map
242                     (into {} (for [key (keys types-map)
243                                    :when (sequential? key)
244                                    :let [type-1 (get types-map key)]
245                                    var key
246                                    :let [type-2 (get types-map var)]]
247                                (if type-2
248                                  [var (list 'and type-1 type-2)]
249                                  [var type-1])))))]
250      (let [types-map (expand-multi-restrictions types-map)]
251        (loop [required lambda-list
252               others ()
253               prefix-rte []
254               suffix-rte []
255               parsed []]
256          (cond (and (empty? required)
257                     (empty? others))
258                ;; finished parsing
259                (if (empty? suffix-rte)
260                  `(:cat ~@prefix-rte)
261                  `(:cat ~@prefix-rte ~@suffix-rte))
262  
263                (and (not-empty required)
264                     (= '& (first required)))
265                ;; found & in the correct place
266                (recur nil ; required
267                       (rest required) ; rest
268                       prefix-rte ; prefix-rte
269                       suffix-rte ; suffix-rte
270                       (conj parsed '&))
271  
272                (and (not-empty others)
273                     (map? (first others)))
274                (letfn [(make-keyword-matcher-rte [var]
275                          (let [k (keyword var)
276                                default-given (contains? (get (first others) :or {}) var)
277                                term-1 (template (:* (:cat (:not (= ~k)) :sigma)))
278                                td (gns/create-and [(get (meta var) :tag :sigma)
279                                                    (get types-map var :sigma)])
280                                term-2 (template (:cat (:* (:cat :sigma :sigma))
281                                                       (:cat (= ~k) ~td)
282                                                       (:* (:cat (:not (= ~k)) :sigma))))]
283                            (list (if default-given
284                                    (template (:or ~term-1
285                                                   ~term-2
286                                                   ))
287                                    term-2))))]
288                  
289                  (assert (vector? (get (first others) :keys))
290                          (cl-format false "parsing ~A expecting a vector specified for :keys, not ~A"
291                                     lambda-list
292                                     (get (first others) :keys)))
293                  
294                  (let [given-keys (get (first others) :keys)
295                        allow-other-keys  (get (first others) :allow-other-keys false)
296                        valid-key (if allow-other-keys
297                                    (template (satisfies keyword?))
298                                    (template (member ~@(map keyword given-keys))))]
299  
300                    (recur required
301                           (rest others)
302                           prefix-rte
303                           (conj suffix-rte
304                                 ;; enforce keyword :sigma pairs
305                                 ;; and simultaneously a constraint for each key specified
306                                 (template (:and (:* (:cat ~valid-key :sigma))
307                                                 ~@(mapcat make-keyword-matcher-rte (:keys (first others))))))
308                           (conj parsed (first others)))))
309                
310                (not-empty required)
311                (let [var (first required)]
312                  (recur (rest required)
313                         others
314                         (conj prefix-rte
315                               (cond (symbol? var)
316                                     ;; parsing required section, found a var
317                                     (gns/create-and [(get (meta var) :tag :sigma)
318                                                      (get types-map var :sigma)])
319                                     (vector? var)
320                                     ;; parsing required section, found a vector
321                                     (list 'rte
322                                           (lambda-list-to-rte (first required) types-map))
323  
324                                     :else ;; found non-symbol non-vector
325                                     (throw (ex-info (cl-format false "invalid lambda-list ~A, at ~A"
326                                                                lambda-list required)
327                                                     {:error-type "cannot parse prefix"
328                                                      :lambda-list lambda-list
329                                                      :parsed parsed
330                                                      :unparsed required}))))
331                         suffix-rte
332                         (conj parsed var)))
333  
334                (and (not-empty others)
335                     (not-empty suffix-rte))
336                (throw (ex-info (cl-format false "invalid lambda-list ~A, at ~A"
337                                           lambda-list others)
338                                {:error-type "cannot parse suffix"
339                                 :lambda-list lambda-list
340                                 :parsed parsed
341                                 :unparsed others}))
342  
343                (and (not-empty others)
344                     (symbol? (first others)))
345                (let [var (first others)]
346                  (recur required
347                         (rest others)
348                         prefix-rte
349                         (conj suffix-rte
350                               (template (:* ~(gns/create-and [(get (meta var) :tag :sigma)
351                                                               (get types-map var :sigma)]))))
352                         (conj parsed var)))
353  
354                (not-empty others)
355                (throw (ex-info (cl-format false "invalid lambda-list ~A, at ~A"
356                                           lambda-list others)
357                                {:error-type "cannot parse suffix"
358                                 :lambda-list lambda-list
359                                 :parsed parsed
360                                 :unparsed others})))))))

380  (defmacro destructuring-case
381    "After evaluating the expression (only once) determine whether its return
382    value conforms to any of the given lambda lists and type restrictions.
383    If so,  bind the variables as if by let, and evaluate the corresponding
384    form.
385    E.g.,
386    (destructuring-case '(true [\"hello\" 3] true)
387       [[_a [_b _c] & _d]  {_a Boolean _b String _d Boolean}]
388       1
389  
390       [[_a _b]          {_a Boolean _b (or String Boolean)}]
391       2)
392  "
393    [expr & operands]
394    (let [[keywords operands] (parse-prefixed-keyword-args operands)]
395      (cond
396        (not= 0 (mod (count operands) 2))
397        (throw (ex-info (cl-format false
398                                   "destructuring-case expects multiple of 2 number of operands after the first:~
399                                  not ~A, ~A"
400                                   (count operands) (apply list 'destructuring-case expr operands))
401                        {:error-type :invalid-destructuring-case-call-site
402                         :expr expr
403                         :operands operands}))
404        
405        :else
406        (let [pairs (partition 2 operands)
407              call-site-meta-data (merge (assoc (meta &form) :file *file*)

581  (defmacro dsfn
582    "Syntactically easier wrapper around destructuring-fn.
583    The syntax of dsfn is the same as the syntax of fn
584    except that meta-data may be placed before a lambda-list which
585    will be considered its constr-map.  E.g.,
586    (dsfn name? (^{x Boolean} [x] ...)
587                (^{x (and Number (not Long))} [x] ...))"
588    [& forms]
589    (letfn [(process [form]
590              (if (and (sequential? form)
591                       (non-empty? form))
592                (let [meta-data (meta (first form))]
593                  (cons [(first form) (if (nil? meta-data) {} meta-data) ] (rest form)))
594                form))]
595      (let [[keywords forms] (parse-prefixed-keyword-args forms)
596            call-site-meta-data (merge (assoc (meta &form) :file *file*)
597                                         keywords)]
598        (cond
599          (and (or (nil? (first forms))
600                   (symbol? (first forms)))
601               (vector? (second forms)))
602          `(dsfn ~@(flatten-1 call-site-meta-data)
603                 ~(first forms) (~@(rest forms)))
604        
605          (vector? (first forms))
606          `(dsfn ~@(flatten-1 call-site-meta-data) (~@forms))
607          
608          :else
609          `(destructuring-fn
610             ~@(flatten-1 call-site-meta-data)
611             ~@(map process forms))))))
