016  (def default-type-vector [:empty-set :sigma ])

020  ;; returns a vector of type designators (each of tyep SimpleTypeD).
021  ;;    this return value is useful in case the caller wants to make
022  ;;    another call to rteToDot making sure the same t0, t1, t2 assignment
023  ;;    is made to each type.  This helps to compare two different
024  ;;    images, so the same type is notated the same way.
025  ;; `habitation` controls whether the AST is colorized into 4 cases
026  ;;    empty/vacuous = red
027  ;;    universe = blue (equiv to Sigma *)
028  ;;    inhabited = green
029  ;;    dont-know = orange
030  (defn rte-to-dot [rte & {:keys [dot-stream 
031                                  title 
032                                  abbrev 
033                                  given-types 
034                                  habitation
035                                  ;; typeLegend controls whether a legend of the types
036                                  ;;    appears in the png file.  otherwise the legend
037                                  ;;    will be printed to stdout
038                                  type-legend
039                                  ]
040                           :or {title ""
041                                abbrev true
042                                given-types default-type-vector
043                                type-legend false
044                                habitation true
045                                }}]
046    (assert rte)
047    (letfn [(linearize [rte]
048              (cond (keyword? rte)
049                    [rte]
050  
051                    (and (sequential? rte)
052                         (keyword? (first rte)))
053                    (distinct (cons rte (mapcat linearize (rest rte))))
054  
055                    :else
056                    [rte]))
057            (children [rte]
058              (cond (keyword? rte)
059                    []
060  
061                    (and (sequential? rte)
062                         (keyword? (first rte)))
063                    (rest rte)
064  
065                    :else
066                    []))
067            (write [input-str]
068              (doseq [c input-str
069                      :let [ i (int c)]
070                      :when (< i 256) ;; silently skip characters out of frange
071                      ]
072                (.write dot-stream (byte (int c)))))
073            (type-label [td tds]
074              (let [idx (get tds td nil)]
075                (assert idx (format "cannot find type %s\nin given merged-types"
076                                    td))
077                (format "t%d" idx)))
078            (node-label [expr tds]
079              (cond (gns/valid-type? expr)
080                    (type-label expr tds)
081  
082                    (rte/valid-rte? expr)
083                    (case expr
084                      (:epsilon) "()"
085                      (:sigma) "&#931;" ;;  Î£
086                      (:empty-set) "{}"
087                      (first expr))))
088            ]
089      
090      (write "digraph G {\n")
091      ;; header
092      (write (format "  # %s\n" rte))
093      (write "  fontname=courier;\n")
094      (write "  rankdir=TB; graph[labeljust=l,nojustify=true]\n")
095      (write "  node [fontname=Arial, fontsize=25];\n")
096      (write "  edge [fontsize=20];\n")
097  
098      (let [linear (linearize rte)
099            used-types (distinct (filter gns/valid-type? linear))
100            node-map (into {} (map-indexed (fn [idx item] [item idx]) linear))
101            ;; collect all the type designators not already in given-types
102            merged-types (concat given-types (filter (fn [td] (not (member td given-types))) used-types ))
103            tds (into {} (map-indexed (fn [idx item] [item idx]) merged-types))
104            ]
105        (doseq [[r idx] node-map
106                :let [label (node-label r tds)]]
107          (write (format "R%s [label=\"%s\"" idx label))
108          (if habitation
109            (do
110              (write ", style=filled, fillcolor=")
111              ;; is r equiv to universe ?,
112              ;; i.e. is !r equiv to emptyset ?
113              (cond (xym/dfa-vacuous? (rte/rte-to-dfa (rte/Not r)) false)
114                    (write "skyblue, shape=egg, orientation=180")
115  
116                    :else
117                    (case (xym/dfa-inhabited? (rte/rte-to-dfa r) :dont-know)
118                      (:dont-know) ;; maybe
119                      (write "orange, shape=Mcircle")
120  
121                      (true) ;; inhabited
122                      (write "lightgreen, shape=rect")
123  
124                      (false)       ;; empty
125                      (write "lightcoral, shape=egg")))))
126          (write "]\n"))
127  
128        (doseq [[parent idx] node-map
129                child (children parent)
130                :let [child-index (node-map child)]]
131          
132          (assert (int? idx))
133          (assert (int? child-index))
134          (write (format "R%d -> R%d\n" idx child-index)))
135  
136  
137        (if (not= "" title)
138          (do (write "  labelloc=\"t\";")
139              (write "\n  label=\"")
140              (write title)
141              (write "\"\n")))
142  
143      (write "}\n")
144      
145      merged-types)))

148                    & {:keys [
149                              title
150                              dir
151                              given-types
152                              dot-file-CB
153                              habitation
154                              ;;// typeLegend controls whether a legend of types
155                              ;;//   appears in the png file.  otherwise the legend will be
156                              ;;//   printed to stdout
157                              type-legend
158                              ]
159                       :or {title ""
160                            dir (ensure-directory (str "/tmp/" (System/getProperty "user.name")))
161                            given-types default-type-vector
162                            dot-file-CB f-dummy
163                            habitation true
164                            type-legend false
165                            }}]
166    (assert rte)
167    (letfn [(to-png [
168                     dot-path-name ;;  String,
169                     _latex-path-name ;; ignored
170                     title ;; : String): Vector[SimpleTypeD] = {
171                     ]
172              (let [dot-stream (java.io.FileOutputStream. (java.io.File. dot-path-name))
173                    merged-types (rte-to-dot rte
174                                             :dot-stream dot-stream
175                                             :dir dir
176                                             :title title
177                                             :given-types given-types
178                                             :habitation habitation
179                                             :type-legend type-legend)]
180                (.close dot-stream)
181                (dot-file-CB dot-path-name)
182                merged-types
183                ))]
